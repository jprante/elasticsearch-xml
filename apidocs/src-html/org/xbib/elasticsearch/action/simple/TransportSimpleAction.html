001package org.xbib.elasticsearch.action.simple;
002
003import org.elasticsearch.action.ActionListener;
004import org.elasticsearch.action.search.SearchRequest;
005import org.elasticsearch.action.search.SearchResponse;
006import org.elasticsearch.action.search.SearchType;
007import org.elasticsearch.action.search.type.TransportSearchCountAction;
008import org.elasticsearch.action.search.type.TransportSearchDfsQueryAndFetchAction;
009import org.elasticsearch.action.search.type.TransportSearchDfsQueryThenFetchAction;
010import org.elasticsearch.action.search.type.TransportSearchQueryAndFetchAction;
011import org.elasticsearch.action.search.type.TransportSearchQueryThenFetchAction;
012import org.elasticsearch.action.search.type.TransportSearchScanAction;
013import org.elasticsearch.action.support.TransportAction;
014import org.elasticsearch.cluster.ClusterService;
015import org.elasticsearch.cluster.ClusterState;
016import org.elasticsearch.common.inject.Inject;
017import org.elasticsearch.common.settings.Settings;
018import org.elasticsearch.indices.IndexMissingException;
019import org.elasticsearch.threadpool.ThreadPool;
020import org.elasticsearch.transport.BaseTransportRequestHandler;
021import org.elasticsearch.transport.TransportChannel;
022import org.elasticsearch.transport.TransportService;
023
024import java.util.Map;
025import java.util.Set;
026
027import static org.elasticsearch.action.search.SearchType.COUNT;
028import static org.elasticsearch.action.search.SearchType.DFS_QUERY_THEN_FETCH;
029import static org.elasticsearch.action.search.SearchType.QUERY_AND_FETCH;
030import static org.elasticsearch.action.search.SearchType.SCAN;
031
032/**
033 * copied from TransportSearchAction
034 */
035public class TransportSimpleAction extends TransportAction<SimpleRequest, SearchResponse> {
036
037    private final ClusterService clusterService;
038    private final TransportSearchDfsQueryThenFetchAction dfsQueryThenFetchAction;
039    private final TransportSearchQueryThenFetchAction queryThenFetchAction;
040    private final TransportSearchDfsQueryAndFetchAction dfsQueryAndFetchAction;
041    private final TransportSearchQueryAndFetchAction queryAndFetchAction;
042    private final TransportSearchScanAction scanAction;
043    private final TransportSearchCountAction countAction;
044    private final boolean optimizeSingleShard;
045
046    @Inject
047    public TransportSimpleAction(Settings settings, ThreadPool threadPool,
048                                 TransportService transportService, ClusterService clusterService,
049                                 TransportSearchDfsQueryThenFetchAction dfsQueryThenFetchAction,
050                                 TransportSearchQueryThenFetchAction queryThenFetchAction,
051                                 TransportSearchDfsQueryAndFetchAction dfsQueryAndFetchAction,
052                                 TransportSearchQueryAndFetchAction queryAndFetchAction,
053                                 TransportSearchScanAction scanAction,
054                                 TransportSearchCountAction countAction) {
055        super(settings, threadPool);
056        this.clusterService = clusterService;
057        this.dfsQueryThenFetchAction = dfsQueryThenFetchAction;
058        this.queryThenFetchAction = queryThenFetchAction;
059        this.dfsQueryAndFetchAction = dfsQueryAndFetchAction;
060        this.queryAndFetchAction = queryAndFetchAction;
061        this.scanAction = scanAction;
062        this.countAction = countAction;
063
064        this.optimizeSingleShard = componentSettings.getAsBoolean("optimize_single_shard", true);
065
066        transportService.registerHandler(SimpleAction.NAME, new TransportHandler());
067    }
068
069    @Override
070    protected void doExecute(SimpleRequest simpleRequest, ActionListener<SearchResponse> listener) {
071        SearchRequest searchRequest = simpleRequest.getSearchRequest();
072        // optimize search type for cases where there is only one shard group to search on
073        if (optimizeSingleShard && searchRequest.searchType() != SCAN && searchRequest.searchType() != COUNT) {
074            try {
075                ClusterState clusterState = clusterService.state();
076                String[] concreteIndices = clusterState.metaData().concreteIndices(searchRequest.indicesOptions(), searchRequest.indices());
077                Map<String, Set<String>> routingMap = clusterState.metaData().resolveSearchRouting(searchRequest.routing(), searchRequest.indices());
078                int shardCount = clusterService.operationRouting().searchShardsCount(clusterState, searchRequest.indices(), concreteIndices, routingMap, searchRequest.preference());
079                if (shardCount == 1) {
080                    // if we only have one group, then we always want Q_A_F, no need for DFS, and no need to do THEN since we hit one shard
081                    searchRequest.searchType(QUERY_AND_FETCH);
082                }
083            } catch (IndexMissingException e) {
084                // ignore this, we will notify the search response if its really the case
085                // from the actual action
086            } catch (Exception e) {
087                logger.debug("failed to optimize search type, continue as normal", e);
088            }
089        }
090
091        if (searchRequest.searchType() == DFS_QUERY_THEN_FETCH) {
092            dfsQueryThenFetchAction.execute(searchRequest, listener);
093        } else if (searchRequest.searchType() == SearchType.QUERY_THEN_FETCH) {
094            queryThenFetchAction.execute(searchRequest, listener);
095        } else if (searchRequest.searchType() == SearchType.DFS_QUERY_AND_FETCH) {
096            dfsQueryAndFetchAction.execute(searchRequest, listener);
097        } else if (searchRequest.searchType() == SearchType.QUERY_AND_FETCH) {
098            queryAndFetchAction.execute(searchRequest, listener);
099        } else if (searchRequest.searchType() == SearchType.SCAN) {
100            scanAction.execute(searchRequest, listener);
101        } else if (searchRequest.searchType() == SearchType.COUNT) {
102            countAction.execute(searchRequest, listener);
103        }
104    }
105
106    private class TransportHandler extends BaseTransportRequestHandler<SimpleRequest> {
107
108        @Override
109        public SimpleRequest newInstance() {
110            return new SimpleRequest();
111        }
112
113        @Override
114        public void messageReceived(SimpleRequest request, final TransportChannel channel) throws Exception {
115            // no need for a threaded listener
116            request.listenerThreaded(false);
117            execute(request, new ActionListener<SearchResponse>() {
118                @Override
119                public void onResponse(SearchResponse result) {
120                    try {
121                        channel.sendResponse(result);
122                    } catch (Throwable e) {
123                        onFailure(e);
124                    }
125                }
126
127                @Override
128                public void onFailure(Throwable e) {
129                    try {
130                        channel.sendResponse(e);
131                    } catch (Exception e1) {
132                        logger.warn("Failed to send response for search", e1);
133                    }
134                }
135            });
136        }
137
138        @Override
139        public String executor() {
140            return ThreadPool.Names.SAME;
141        }
142    }
143
