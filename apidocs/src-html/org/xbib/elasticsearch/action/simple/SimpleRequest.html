001package org.xbib.elasticsearch.action.simple;
002
003import org.elasticsearch.action.ActionRequest;
004import org.elasticsearch.action.ActionRequestValidationException;
005import org.elasticsearch.action.search.SearchRequest;
006import org.elasticsearch.common.io.stream.StreamInput;
007import org.elasticsearch.common.io.stream.StreamOutput;
008
009import java.io.IOException;
010
011public class SimpleRequest extends ActionRequest<SimpleRequest> {
012
013    private SearchRequest searchRequest;
014
015    public void setSearchRequest(SearchRequest searchRequest) {
016        this.searchRequest = searchRequest;
017    }
018
019    public SearchRequest getSearchRequest() {
020        return searchRequest;
021    }
022
023    @Override
024    public ActionRequestValidationException validate() {
025        return null;
026    }
027
028    @Override
029    public void readFrom(StreamInput in) throws IOException {
030        super.readFrom(in);
031        searchRequest = new SearchRequest();
032        searchRequest.readFrom(in);
033    }
034
035    @Override
036    public void writeTo(StreamOutput out) throws IOException {
037        super.writeTo(out);
038        searchRequest.writeTo(out);
039    }
040}
