002    package org.xbib.elasticsearch.rest.xml;
003    
004    import org.elasticsearch.ElasticsearchException;
005    import org.elasticsearch.common.component.AbstractLifecycleComponent;
006    import org.elasticsearch.common.inject.Inject;
007    import org.elasticsearch.common.settings.Settings;
008    import org.elasticsearch.rest.RestController;
009    
010    public class XmlService extends AbstractLifecycleComponent<XmlService> {
011    
012        private final RestController controller;
013    
014        @Inject
015        public XmlService(Settings settings, RestController controller) {
016            super(settings);
017            this.controller = controller;
018        }
019    
020        @Override
021        protected void doStart() throws ElasticsearchException {
022            controller.registerFilter(new XmlFilter());
023        }
024    
025        @Override
026        protected void doStop() throws ElasticsearchException {
027        }
028    
029        @Override
030        protected void doClose() throws ElasticsearchException {
031        }
032    }
