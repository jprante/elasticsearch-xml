001package org.xbib.elasticsearch.rest.xml;
002
003import org.elasticsearch.common.bytes.BytesReference;
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.elasticsearch.common.xcontent.XContentBuilder;
007import org.elasticsearch.common.xcontent.XContentFactory;
008import org.elasticsearch.common.xcontent.XContentType;
009import org.elasticsearch.common.xcontent.XContentParser;
010import org.elasticsearch.http.HttpChannel;
011import org.elasticsearch.http.HttpRequest;
012import org.elasticsearch.rest.BytesRestResponse;
013import org.elasticsearch.rest.RestChannel;
014import org.elasticsearch.rest.RestFilter;
015import org.elasticsearch.rest.RestFilterChain;
016import org.elasticsearch.rest.RestRequest;
017import org.elasticsearch.rest.RestResponse;
018import org.elasticsearch.rest.RestStatus;
019
020import org.xbib.elasticsearch.common.xcontent.XmlXContentBuilder;
021import org.xbib.elasticsearch.common.xcontent.XmlXContentFactory;
022import org.xbib.elasticsearch.common.xcontent.XmlXContentType;
023import org.xbib.elasticsearch.common.xcontent.xml.XmlXParams;
024
025import java.util.Map;
026
027/**
028 * XML filter for Elasticsearch REST requests and responses.
029 *
030 * To receive XML responses, the request header Accept: must contain "application/xml"
031 */
032public class XmlFilter extends RestFilter {
033
034    private final ESLogger logger = ESLoggerFactory.getLogger(XmlFilter.class.getName());
035
036    private final XmlXParams params;
037
038    public XmlFilter() {
039        this.params = new XmlXParams();
040    }
041
042    @Override
043    public void process(RestRequest request, RestChannel channel, RestFilterChain filterChain) {
044        filterChain.continueProcessing(new XmlRequest(request), new XmlChannel(request, channel));
045    }
046
047    private boolean isXml(RestRequest request) {
048        return "application/xml".equals(request.header("Accept"))
049                || request.hasParam("xml");
050    }
051
052    /**
053     * Unwraps an XML REST request to JSON if Content-type: header declares application/xml.
054     * We must extend HttpRequest because this will get used in a casting in the HTTP controller.
055     */
056    class XmlRequest extends HttpRequest {
057
058        private RestRequest request;
059
060        XmlRequest(RestRequest request) {
061            this.request = request;
062        }
063
064        @Override
065        public Method method() {
066            return request.method();
067        }
068
069        @Override
070        public String uri() {
071            return request.uri();
072        }
073
074        @Override
075        public String rawPath() {
076            return request.rawPath();
077        }
078
079        @Override
080        public boolean hasContent() {
081            return request.hasContent();
082        }
083
084        @Override
085        public boolean contentUnsafe() {
086            return request.contentUnsafe();
087        }
088
089        @Override
090        public BytesReference content() {
091            if (isXml(request)) {
092                XContentParser parser = null;
093                try {
094                    BytesReference b = request.content();
095                    parser = XmlXContentFactory.xContent(XmlXContentType.XML).createParser(b);
096                    parser.nextToken();
097                    XContentBuilder builder = XContentFactory.jsonBuilder();
098                    builder.copyCurrentStructure(parser);
099                    return builder.bytes();
100                } catch (Throwable e) {
101                    logger.error(e.getMessage(), e);
102                } finally {
103                    if (parser != null) {
104                        parser.close();
105                    }
106                }
107            }
108            return request.content();
109        }
110
111        @Override
112        public String header(String name) {
113            return request.header(name);
114        }
115
116        @Override
117        public Iterable<Map.Entry<String, String>> headers() {
118            return request.headers();
119        }
120
121        @Override
122        public boolean hasParam(String key) {
123            return false;
124        }
125
126        @Override
127        public String param(String key) {
128            return request.param(key);
129        }
130
131        @Override
132        public String param(String key, String defaultValue) {
133            return request.param(key, defaultValue);
134        }
135
136        @Override
137        public Map<String, String> params() {
138            return request.params();
139        }
140    }
141
142    /**
143     * Wraps a REST channel response into XML if Accept: header declares application/xml.
144     * This must extend HttpChannel because this will get used in a casting in the HTTP controller.
145     */
146    class XmlChannel extends HttpChannel {
147
148        private final RestChannel channel;
149
150        XmlChannel(RestRequest request, RestChannel channel) {
151            super(request);
152            this.channel = channel;
153        }
154
155        @Override
156        public void sendResponse(RestResponse response) {
157            if (!response.status().equals(RestStatus.OK)) {
158                channel.sendResponse(response);
159            }
160            if (isXml(request)) {
161                XContentParser parser = null;
162                try {
163                    String string = response.content().toUtf8(); // takes some space ... :(
164                    XContentType xContentType = XContentFactory.xContentType(string);
165                    parser = XContentFactory.xContent(xContentType).createParser(string);
166                    parser.nextToken();
167                    XmlXContentBuilder builder = XmlXContentFactory.xmlBuilder(params);
168                    if (request.paramAsBoolean("pretty", false)) {
169                        builder.prettyPrint();
170                    }
171                    builder.copyCurrentStructure(parser);
172                    channel.sendResponse(new BytesRestResponse(RestStatus.OK, "text/xml; charset=UTF-8", builder.bytes(), false));
173                    return;
174                } catch (Throwable e) {
175                    logger.error(e.getMessage(), e);
176                    channel.sendResponse(new BytesRestResponse(RestStatus.INTERNAL_SERVER_ERROR, e.getMessage()));
177                    return;
178                } finally {
179                    if (parser != null) {
180                        parser.close();
181                    }
182                }
183            }
184            channel.sendResponse(response);
185        }
186    }
187
188}
