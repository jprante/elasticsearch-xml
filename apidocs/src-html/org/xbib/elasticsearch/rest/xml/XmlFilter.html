031    /**
032     * XML filter for Elasticsearch REST requests and responses.
033     *
034     * To receive XML responses, the request header Accept: must contain "application/xml"
035     */
036    public class XmlFilter extends RestFilter {
037    
038        private final ESLogger logger = ESLoggerFactory.getLogger(XmlFilter.class.getName());
039    
040        private final XmlXParams params;
041    
042        public XmlFilter() {
043            this.params = new XmlXParams();
044        }
045    
046        @Override
047        public void process(RestRequest request, RestChannel channel, RestFilterChain filterChain) {
048            filterChain.continueProcessing(new XmlRequest(request), new XmlChannel(request, channel));
049        }
050    
051        private boolean isXml(RestRequest request) {
052            return "application/xml".equals(request.header("Accept"))
053                    || request.hasParam("xml");
054        }
055    
056        /**
057         * Unwraps an XML REST request to JSON if Content-type: header declares application/xml.
058         * We must extend HttpRequest because this will get used in a casting in the HTTP controller.
059         */
060        class XmlRequest extends HttpRequest {
061    
062            private RestRequest request;
063    
064            XmlRequest(RestRequest request) {
065                this.request = request;
066            }
067    
068            @Override
069            public Method method() {
070                return request.method();
071            }
072    
073            @Override
074            public String uri() {
075                return request.uri();
076            }
077    
078            @Override
079            public String rawPath() {
080                return request.rawPath();
081            }
082    
083            @Override
084            public boolean hasContent() {
085                return request.hasContent();
086            }
087    
088            @Override
089            public boolean contentUnsafe() {
090                return request.contentUnsafe();
091            }
092    
093            @Override
094            public BytesReference content() {
095                if (isXml(request)) {
096                    XContentParser parser = null;
097                    try {
098                        BytesReference b = request.content();
099                        parser = XmlXContentFactory.xContent(XmlXContentType.XML).createParser(b);
100                        parser.nextToken();
101                        XContentBuilder builder = XContentFactory.jsonBuilder();
102                        builder.copyCurrentStructure(parser);
103                        return builder.bytes();
104                    } catch (Throwable e) {
105                        logger.error(e.getMessage(), e);
106                    } finally {
107                        if (parser != null) {
108                            parser.close();
109                        }
110                    }
111                }
112                return request.content();
113            }
114    
115            @Override
116            public String header(String name) {
117                return request.header(name);
118            }
119    
120            @Override
121            public Iterable<Map.Entry<String, String>> headers() {
122                return request.headers();
123            }
124    
125            @Override
126            public boolean hasParam(String key) {
127                return false;
128            }
129    
130            @Override
131            public String param(String key) {
132                return request.param(key);
133            }
134    
135            @Override
136            public String param(String key, String defaultValue) {
137                return request.param(key, defaultValue);
138            }
139    
140            @Override
141            public Map<String, String> params() {
142                return request.params();
143            }
144        }
145    
146        /**
147         * Wraps a REST channel response into XML if Accept: header declares application/xml.
148         * This must extend HttpChannel because this will get used in a casting in the HTTP controller.
149         */
150        class XmlChannel implements HttpChannel {
151    
152            private final RestRequest request;
153    
154            private final RestChannel channel;
155    
156            XmlChannel(RestRequest request, RestChannel channel) {
157                this.request = request;
158                this.channel = channel;
159            }
160    
161            @Override
162            public void sendResponse(RestResponse response) {
163                if (!response.status().equals(RestStatus.OK)) {
164                    channel.sendResponse(response);
165                }
166                if (isXml(request)) {
167                    XContentParser parser = null;
168                    try {
169                        byte[] b = response.content();
170                        XContentType xContentType = XContentFactory.xContentType(b);
171                        parser = XContentFactory.xContent(xContentType).createParser(b);
172                        parser.nextToken();
173                        XmlXContentBuilder builder = XmlXContentFactory.xmlBuilder(params);
174                        if (request.paramAsBoolean("pretty", false)) {
175                            builder.prettyPrint();
176                        }
177                        builder.copyCurrentStructure(parser);
178                        channel.sendResponse(new XmlRestResponse(RestStatus.OK, builder.string()));
179                        return;
180                    } catch (Throwable e) {
181                        logger.error(e.getMessage(), e);
182                        channel.sendResponse(new StringRestResponse(RestStatus.INTERNAL_SERVER_ERROR, e.getMessage()));
183                        return;
184                    } finally {
185                        if (parser != null) {
186                            parser.close();
187                        }
188                    }
189                }
190                channel.sendResponse(response);
191            }
192        }
193    
194        static class XmlRestResponse extends Utf8RestResponse {
195    
196            public XmlRestResponse(RestStatus status, String content) {
197                super(status, convert(content));
198            }
199    
200            public String contentType() {
201                return "text/xml; charset=UTF-8";
202            }
203    
204            private static BytesRef convert(String content) {
205                BytesRef result = new BytesRef();
206                UnicodeUtil.UTF16toUTF8(content, 0, content.length(), result);
207                return result;
208            }
209    
210        }
211    }
