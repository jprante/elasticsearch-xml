225        public XmlXContentBuilder field(XContentBuilderString name) throws IOException {
226            if (fieldCaseConversion == FieldCaseConversion.UNDERSCORE) {
227                generator.writeFieldName(name.underscore());
228            } else if (fieldCaseConversion == FieldCaseConversion.CAMELCASE) {
229                generator.writeFieldName(name.camelCase());
230            } else {
231                generator.writeFieldName(name.underscore());
232            }
233            return this;
234        }

247        public XmlXContentBuilder field(String name) throws IOException {
248            if (fieldCaseConversion == FieldCaseConversion.UNDERSCORE) {
249                if (cachedStringBuilder == null) {
250                    cachedStringBuilder = new StringBuilder();
251                }
252                name = Strings.toUnderscoreCase(name, cachedStringBuilder);
253            } else if (fieldCaseConversion == FieldCaseConversion.CAMELCASE) {
254                if (cachedStringBuilder == null) {
255                    cachedStringBuilder = new StringBuilder();
256                }
257                name = Strings.toCamelCase(name, cachedStringBuilder);
258            }
259            generator.writeFieldName(name);
260            return this;
261        }

527        public XmlXContentBuilder field(String name, Text value) throws IOException {
528            field(name);
529            if (value.hasBytes() && value.bytes().hasArray()) {
530                generator.writeUTF8String(value.bytes().array(), value.bytes().arrayOffset(), value.bytes().length());
531                return this;
532            }
533            if (value.hasString()) {
534                generator.writeString(value.string());
535                return this;
536            }
537            // TODO: TextBytesOptimization we can use a buffer here to convert it? maybe add a request to jackson to support InputStream as well?
538            BytesArray bytesArray = value.bytes().toBytesArray();
539            generator.writeUTF8String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length());
540            return this;
541        }

1079        private void writeMap(Map<String, Object> map) throws IOException {
1080            generator.writeStartObject();
1081    
1082            for (Map.Entry<String, Object> entry : map.entrySet()) {
1083                field(entry.getKey());
1084                Object value = entry.getValue();
1085                if (value == null) {
1086                    generator.writeNull();
1087                } else {
1088                    writeValue(value);
1089                }
1090            }
1091            generator.writeEndObject();
1092        }
1093    
1094        private void writeValue(Object value) throws IOException {
1095            if (value == null) {
1096                generator.writeNull();
1097                return;
1098            }
1099            Class type = value.getClass();
1100            if (type == String.class) {
1101                generator.writeString((String) value);
1102            } else if (type == Integer.class) {
1103                generator.writeNumber(((Integer) value).intValue());
1104            } else if (type == Long.class) {
1105                generator.writeNumber(((Long) value).longValue());
1106            } else if (type == Float.class) {
1107                generator.writeNumber(((Float) value).floatValue());
1108            } else if (type == Double.class) {
1109                generator.writeNumber(((Double) value).doubleValue());
1110            } else if (type == Short.class) {
1111                generator.writeNumber(((Short) value).shortValue());
1112            } else if (type == Boolean.class) {
1113                generator.writeBoolean(((Boolean) value).booleanValue());
1114            } else if (type == GeoPoint.class) {
1115                generator.writeStartObject();
1116                generator.writeNumberField("lat", ((GeoPoint) value).lat());
1117                generator.writeNumberField("lon", ((GeoPoint) value).lon());
1118                generator.writeEndObject();
1119            } else if (value instanceof Map) {
1120                writeMap((Map) value);
1121            } else if (value instanceof Iterable) {
1122                generator.writeStartArray();
1123                for (Object v : (Iterable) value) {
1124                    writeValue(v);
1125                }
1126                generator.writeEndArray();
1127            } else if (value instanceof Object[]) {
1128                generator.writeStartArray();
1129                for (Object v : (Object[]) value) {
1130                    writeValue(v);
1131                }
1132                generator.writeEndArray();
1133            } else if (type == byte[].class) {
1134                generator.writeBinary((byte[]) value);
1135            } else if (value instanceof Date) {
1136                generator.writeString(XmlXContentBuilder.defaultDatePrinter.print(((Date) value).getTime()));
1137            } else if (value instanceof Calendar) {
1138                generator.writeString(XmlXContentBuilder.defaultDatePrinter.print((((Calendar) value)).getTimeInMillis()));
1139            } else if (value instanceof ReadableInstant) {
1140                generator.writeString(XmlXContentBuilder.defaultDatePrinter.print((((ReadableInstant) value)).getMillis()));
1141            } else if (value instanceof BytesReference) {
1142                BytesReference bytes = (BytesReference) value;
1143                if (!bytes.hasArray()) {
1144                    bytes = bytes.toBytesArray();
1145                }
1146                generator.writeBinary(bytes.array(), bytes.arrayOffset(), bytes.length());
1147            } else if (value instanceof Text) {
1148                Text text = (Text) value;
1149                if (text.hasBytes() && text.bytes().hasArray()) {
1150                    generator.writeUTF8String(text.bytes().array(), text.bytes().arrayOffset(), text.bytes().length());
1151                } else if (text.hasString()) {
1152                    generator.writeString(text.string());
1153                } else {
1154                    BytesArray bytesArray = text.bytes().toBytesArray();
1155                    generator.writeUTF8String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length());
1156                }
1157            //} else if (value instanceof ToXContent) {
1158            //    ((ToXContent) value).toXContent(this, ToXContent.EMPTY_PARAMS);
1159            } else if (value instanceof double[]) {
1160                generator.writeStartArray();
1161                for (double v : (double[]) value) {
1162                    generator.writeNumber(v);
1163                }
1164                generator.writeEndArray();
1165            } else if (value instanceof long[]) {
1166                generator.writeStartArray();
1167                for (long v : (long[]) value) {
1168                    generator.writeNumber(v);
1169                }
1170                generator.writeEndArray();
1171            } else if (value instanceof int[]) {
1172                generator.writeStartArray();
1173                for (int v : (int[]) value) {
1174                    generator.writeNumber(v);
1175                }
1176                generator.writeEndArray();
1177            } else if (value instanceof float[]) {
1178                generator.writeStartArray();
1179                for (float v : (float[]) value) {
1180                    generator.writeNumber(v);
1181                }
1182                generator.writeEndArray();
1183            } else if (value instanceof short[]) {
1184                generator.writeStartArray();
1185                for (float v : (short[]) value) {
1186                    generator.writeNumber(v);
1187                }
1188                generator.writeEndArray();
1189            } else {
1190                // if this is a "value" object, like enum, DistanceUnit, ..., just toString it
1191                // yea, it can be misleading when toString a Java class, but really, jackson should be used in that case
1192                generator.writeString(value.toString());
1193                //throw new ElasticsearchIllegalArgumentException("type not supported for generic value conversion: " + type);
1194            }
1195        }
