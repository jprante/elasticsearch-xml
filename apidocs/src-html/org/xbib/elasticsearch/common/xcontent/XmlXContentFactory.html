086        /**
087         * Constructs a xcontent builder that will output the result into the provided output stream.
088         */
089        public static XmlXContentBuilder contentBuilder(XmlXContentType type, OutputStream outputStream) throws IOException {
090            if (type == XmlXContentType.JSON) {
091                return jsonBuilder(outputStream);
092            } else if (type == XmlXContentType.SMILE) {
093                return smileBuilder(outputStream);
094            } else if (type == XmlXContentType.YAML) {
095                return yamlBuilder(outputStream);
096            } else if (type == XmlXContentType.XML) {
097                return xmlBuilder(outputStream);
098            }
099            throw new ElasticsearchIllegalArgumentException("No matching content type for " + type);
100        }
101    
102        /**
103         * Returns a binary content builder for the provided content type.
104         */
105        public static XmlXContentBuilder contentBuilder(XmlXContentType type) throws IOException {
106            if (type == XmlXContentType.XML) {
107                XmlXParams params = new XmlXParams();
108                return XmlXContent.contentBuilder(params);
109            }
110            throw new ElasticsearchIllegalArgumentException("No matching content type for " + type);
111        }

120        /**
121         * Guesses the content type based on the provided char sequence.
122         */
123        public static XmlXContentType xContentType(CharSequence content) {
124            int length = content.length() < GUESS_HEADER_LENGTH ? content.length() : GUESS_HEADER_LENGTH;
125            if (length == 0) {
126                return null;
127            }
128            char first = content.charAt(0);
129            if (first == '{') {
130                return XmlXContentType.JSON;
131            }
132            // Should we throw a failure here? Smile idea is to use it in bytes....
133            if (length > 2 && first == SmileConstants.HEADER_BYTE_1 && content.charAt(1) == SmileConstants.HEADER_BYTE_2 && content.charAt(2) == SmileConstants.HEADER_BYTE_3) {
134                return XmlXContentType.SMILE;
135            }
136            if (length > 2 && first == '-' && content.charAt(1) == '-' && content.charAt(2) == '-') {
137                return XmlXContentType.YAML;
138            }
139            for (int i = 0; i < length; i++) {
140                char c = content.charAt(i);
141                if (c == '{') {
142                    return XmlXContentType.JSON;
143                }
144                if (c == '<') {
145                    return XmlXContentType.XML;
146                }
147            }
148            return null;
149        }

151        /**
152         * Guesses the content (type) based on the provided char sequence.
153         */
154        public static XContent xContent(CharSequence content) {
155            XmlXContentType type = xContentType(content);
156            if (type == null) {
157                throw new ElasticsearchParseException("Failed to derive xcontent from " + content);
158            }
159            return xContent(type);
160        }

187        /**
188         * Guesses the content type based on the provided input stream.
189         */
190        public static XmlXContentType xContentType(InputStream si) throws IOException {
191            int first = si.read();
192            if (first == -1) {
193                return null;
194            }
195            int second = si.read();
196            if (second == -1) {
197                return null;
198            }
199            if (first == SmileConstants.HEADER_BYTE_1 && second == SmileConstants.HEADER_BYTE_2) {
200                int third = si.read();
201                if (third == SmileConstants.HEADER_BYTE_3) {
202                    return XmlXContentType.SMILE;
203                }
204            }
205            if (first == '{' || second == '{') {
206                return XmlXContentType.JSON;
207            }
208            if (first == '-' && second == '-') {
209                int third = si.read();
210                if (third == '-') {
211                    return XmlXContentType.YAML;
212                }
213            }
214            if (first == '<' && second == '?') {
215                int third = si.read();
216                if (third == 'x') {
217                    return XmlXContentType.XML;
218                }
219            }
220            for (int i = 2; i < GUESS_HEADER_LENGTH; i++) {
221                int val = si.read();
222                if (val == -1) {
223                    return null;
224                }
225                if (val == '{') {
226                    return XmlXContentType.JSON;
227                }
228            }
229            return null;
230        }

240            XmlXContentType type = xContentType(bytes);

243            }
244            return xContent(type);
245        }

247        /**
248         * Guesses the content type based on the provided bytes.
249         */
250        public static XmlXContentType xContentType(BytesReference bytes) {
251            int length = bytes.length() < GUESS_HEADER_LENGTH ? bytes.length() : GUESS_HEADER_LENGTH;
252            if (length == 0) {
253                return null;
254            }
255            byte first = bytes.get(0);
256            if (first == '{') {
257                return XmlXContentType.JSON;
258            }
259            if (length > 2 && first == SmileConstants.HEADER_BYTE_1 && bytes.get(1) == SmileConstants.HEADER_BYTE_2 && bytes.get(2) == SmileConstants.HEADER_BYTE_3) {
260                return XmlXContentType.SMILE;
261            }
262            if (length > 2 && first == '-' && bytes.get(1) == '-' && bytes.get(2) == '-') {
263                return XmlXContentType.YAML;
264            }
265            if (length > 2 && first == '<' && bytes.get(1) == '?' && bytes.get(2) == 'x') {
266                return XmlXContentType.XML;
267            }
268            for (int i = 0; i < length; i++) {
269                if (bytes.get(i) == '{') {
270                    return XmlXContentType.JSON;
271                }
272            }
273            return null;
274        }
