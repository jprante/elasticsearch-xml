103    /**
104     * Constructs a xcontent builder that will output the result into the provided output stream.
105     */
106    public static XmlXContentBuilder contentBuilder(XmlXContentType type, OutputStream outputStream) throws IOException {
107        if (type == XmlXContentType.JSON) {
108            return jsonBuilder(outputStream);
109        } else if (type == XmlXContentType.SMILE) {
110            return smileBuilder(outputStream);
111        } else if (type == XmlXContentType.YAML) {
112            return yamlBuilder(outputStream);
113        } else if (type == XmlXContentType.XML) {
114            return xmlBuilder(outputStream);
115        }
116        throw new ElasticsearchIllegalArgumentException("No matching content type for " + type);
117    }

119    /**
120     * Returns a binary content builder for the provided content type.
121     */
122    public static XmlXContentBuilder contentBuilder(XmlXContentType type) throws IOException {
123        if (type == XmlXContentType.XML) {
124            XmlXParams params = new XmlXParams();
125            return XmlXContent.contentBuilder(params);
126        }
127        throw new ElasticsearchIllegalArgumentException("No matching content type for " + type);
128    }

137    /**
138     * Guesses the content type based on the provided char sequence.
139     */
140    public static XmlXContentType xContentType(CharSequence content) {
141        int length = content.length() < GUESS_HEADER_LENGTH ? content.length() : GUESS_HEADER_LENGTH;
142        if (length == 0) {
143            return null;
144        }
145        char first = content.charAt(0);
146        if (first == '{') {
147            return XmlXContentType.JSON;
148        }
149        // Should we throw a failure here? Smile idea is to use it in bytes....
150        if (length > 2 && first == SmileConstants.HEADER_BYTE_1 && content.charAt(1) == SmileConstants.HEADER_BYTE_2 && content.charAt(2) == SmileConstants.HEADER_BYTE_3) {
151            return XmlXContentType.SMILE;
152        }
153        if (length > 2 && first == '-' && content.charAt(1) == '-' && content.charAt(2) == '-') {
154            return XmlXContentType.YAML;
155        }
156        for (int i = 0; i < length; i++) {
157            char c = content.charAt(i);
158            if (c == '{') {
159                return XmlXContentType.JSON;
160            }
161            if (c == '<') {
162                return XmlXContentType.XML;
163            }
164        }
165        return null;
166    }
167
168    /**
169     * Guesses the content (type) based on the provided char sequence.
170     */
171    public static XContent xContent(CharSequence content) {
172        XmlXContentType type = xContentType(content);
173        if (type == null) {
174            throw new ElasticsearchParseException("Failed to derive xcontent from " + content);
175        }
176        return xContent(type);
177    }

204    /**
205     * Guesses the content type based on the provided input stream.
206     */
207    public static XmlXContentType xContentType(InputStream si) throws IOException {
208        int first = si.read();
209        if (first == -1) {
210            return null;
211        }
212        int second = si.read();
213        if (second == -1) {
214            return null;
215        }
216        if (first == SmileConstants.HEADER_BYTE_1 && second == SmileConstants.HEADER_BYTE_2) {
217            int third = si.read();
218            if (third == SmileConstants.HEADER_BYTE_3) {
219                return XmlXContentType.SMILE;
220            }
221        }
222        if (first == '{' || second == '{') {
223            return XmlXContentType.JSON;
224        }
225        if (first == '-' && second == '-') {
226            int third = si.read();
227            if (third == '-') {
228                return XmlXContentType.YAML;
229            }
230        }
231        if (first == '<' && second == '?') {
232            int third = si.read();
233            if (third == 'x') {
234                return XmlXContentType.XML;
235            }
236        }
237        for (int i = 2; i < GUESS_HEADER_LENGTH; i++) {
238            int val = si.read();
239            if (val == -1) {
240                return null;
241            }
242            if (val == '{') {
243                return XmlXContentType.JSON;
244            }
245        }
246        return null;
247    }

253        return xContentType(new BytesArray(data, offset, length));

256    public static XContent xContent(BytesReference bytes) {
257        XmlXContentType type = xContentType(bytes);
258        if (type == null) {
259            throw new ElasticsearchParseException("Failed to derive xcontent from " + bytes);
260        }
261        return xContent(type);
262    }
263
264    /**
265     * Guesses the content type based on the provided bytes.
266     */
267    public static XmlXContentType xContentType(BytesReference bytes) {
268        int length = bytes.length() < GUESS_HEADER_LENGTH ? bytes.length() : GUESS_HEADER_LENGTH;
269        if (length == 0) {
270            return null;
271        }
272        byte first = bytes.get(0);
273        if (first == '{') {
274            return XmlXContentType.JSON;
275        }
276        if (length > 2 && first == SmileConstants.HEADER_BYTE_1 && bytes.get(1) == SmileConstants.HEADER_BYTE_2 && bytes.get(2) == SmileConstants.HEADER_BYTE_3) {
277            return XmlXContentType.SMILE;
278        }
279        if (length > 2 && first == '-' && bytes.get(1) == '-' && bytes.get(2) == '-') {
280            return XmlXContentType.YAML;
281        }
282        if (length > 2 && first == '<' && bytes.get(1) == '?' && bytes.get(2) == 'x') {
283            return XmlXContentType.XML;
284        }
285        for (int i = 0; i < length; i++) {
286            if (bytes.get(i) == '{') {
287                return XmlXContentType.JSON;
288            }
289        }
290        return null;
291    }
