047    public static XContentParser createParser(BytesReference bytes) throws IOException {
048        if (bytes.hasArray()) {
049            return createParser(bytes.array(), bytes.arrayOffset(), bytes.length());
050        }
051        Compressor compressor = CompressorFactory.compressor(bytes);
052        if (compressor != null) {
053            CompressedStreamInput compressedInput = compressor.streamInput(bytes.streamInput());
054            XmlXContentType contentType = XmlXContentFactory.xContentType(compressedInput);
055            compressedInput.resetToBufferStart();
056            return XmlXContentFactory.xContent(contentType).createParser(compressedInput);
057        } else {
058            return XmlXContentFactory.xContent(bytes).createParser(bytes.streamInput());
059        }
060    }
061
062    public static XContentParser createParser(byte[] data, int offset, int length) throws IOException {
063        Compressor compressor = CompressorFactory.compressor(data, offset, length);
064        if (compressor != null) {
065            CompressedStreamInput compressedInput = compressor.streamInput(new BytesStreamInput(data, offset, length, false));
066            XmlXContentType contentType = XmlXContentFactory.xContentType(compressedInput);
067            compressedInput.resetToBufferStart();
068            return XmlXContentFactory.xContent(contentType).createParser(compressedInput);
069        } else {
070            return XmlXContentFactory.xContent(data, offset, length).createParser(data, offset, length);
071        }
072    }

074    public static Tuple<XmlXContentType, Map<String, Object>> convertToMap(BytesReference bytes, boolean ordered)
075            throws ElasticsearchParseException {
076        if (bytes.hasArray()) {
077            return convertToMap(bytes.array(), bytes.arrayOffset(), bytes.length(), ordered);
078        }
079        try {
080            XContentParser parser;
081            XmlXContentType contentType;
082            Compressor compressor = CompressorFactory.compressor(bytes);
083            if (compressor != null) {
084                CompressedStreamInput compressedStreamInput = compressor.streamInput(bytes.streamInput());
085                contentType = XmlXContentFactory.xContentType(compressedStreamInput);
086                compressedStreamInput.resetToBufferStart();
087                parser = XmlXContentFactory.xContent(contentType).createParser(compressedStreamInput);
088            } else {
089                contentType = XmlXContentFactory.xContentType(bytes);
090                parser = XmlXContentFactory.xContent(contentType).createParser(bytes.streamInput());
091            }
092            if (ordered) {
093                return Tuple.tuple(contentType, parser.mapOrderedAndClose());
094            } else {
095                return Tuple.tuple(contentType, parser.mapAndClose());
096            }
097        } catch (IOException e) {
098            throw new ElasticsearchParseException("Failed to parse content to map", e);
099        }
100    }

136    public static String convertToJson(BytesReference bytes, boolean reformatJson, boolean prettyPrint) throws IOException {
137        if (bytes.hasArray()) {
138            return convertToJson(bytes.array(), bytes.arrayOffset(), bytes.length(), reformatJson, prettyPrint);
139        }
140        XmlXContentType xmlXContentType = XmlXContentFactory.xContentType(bytes);
141        if (xmlXContentType == XmlXContentType.JSON && !reformatJson) {
142            BytesArray bytesArray = bytes.toBytesArray();
143            return new String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length(), Charsets.UTF_8);
144        }
145        XContentParser parser = null;
146        try {
147            parser = XmlXContentFactory.xContent(xmlXContentType).createParser(bytes.streamInput());
148            parser.nextToken();
149            XmlXContentBuilder builder = XmlXContentFactory.jsonBuilder();
150            if (prettyPrint) {
151                builder.prettyPrint();
152            }
153            builder.copyCurrentStructure(parser);
154            return builder.string();
155        } finally {
156            if (parser != null) {
157                parser.close();
158            }
159        }
160    }

211    /**
212     * Updates the provided changes into the source. If the key exists in the changes, it overrides the one in source
213     * unless both are Maps, in which case it recuersively updated it.
214     */
215    public static void update(Map<String, Object> source, Map<String, Object> changes) {
216        for (Map.Entry<String, Object> changesEntry : changes.entrySet()) {
217            if (!source.containsKey(changesEntry.getKey())) {
218                // safe to copy, change does not exist in source
219                source.put(changesEntry.getKey(), changesEntry.getValue());
220            } else {
221                if (source.get(changesEntry.getKey()) instanceof Map && changesEntry.getValue() instanceof Map) {
222                    // recursive merge maps
223                    update((Map<String, Object>) source.get(changesEntry.getKey()), (Map<String, Object>) changesEntry.getValue());
224                } else {
225                    // update the field
226                    source.put(changesEntry.getKey(), changesEntry.getValue());
227                }
228            }
229        }
230    }

232    /**
233     * Merges the defaults provided as the second parameter into the content of the first. Only does recursive merge
234     * for inner maps.
235     */
236    @SuppressWarnings({"unchecked"})
237    public static void mergeDefaults(Map<String, Object> content, Map<String, Object> defaults) {
238        for (Map.Entry<String, Object> defaultEntry : defaults.entrySet()) {
239            if (!content.containsKey(defaultEntry.getKey())) {
240                // copy it over, it does not exists in the content
241                content.put(defaultEntry.getKey(), defaultEntry.getValue());
242            } else {
243                // in the content and in the default, only merge compound ones (maps)
244                if (content.get(defaultEntry.getKey()) instanceof Map && defaultEntry.getValue() instanceof Map) {
245                    mergeDefaults((Map<String, Object>) content.get(defaultEntry.getKey()), (Map<String, Object>) defaultEntry.getValue());
246                } else if (content.get(defaultEntry.getKey()) instanceof List && defaultEntry.getValue() instanceof List) {
247                    List defaultList = (List) defaultEntry.getValue();
248                    List contentList = (List) content.get(defaultEntry.getKey());
249
250                    List mergedList = new ArrayList();
251                    if (allListValuesAreMapsOfOne(defaultList) && allListValuesAreMapsOfOne(contentList)) {
252                        // all are in the form of [ {"key1" : {}}, {"key2" : {}} ], merge based on keys
253                        Map<String, Map<String, Object>> processed = Maps.newLinkedHashMap();
254                        for (Object o : contentList) {
255                            Map<String, Object> map = (Map<String, Object>) o;
256                            Map.Entry<String, Object> entry = map.entrySet().iterator().next();
257                            processed.put(entry.getKey(), map);
258                        }
259                        for (Object o : defaultList) {
260                            Map<String, Object> map = (Map<String, Object>) o;
261                            Map.Entry<String, Object> entry = map.entrySet().iterator().next();
262                            if (processed.containsKey(entry.getKey())) {
263                                mergeDefaults(processed.get(entry.getKey()), map);
264                            }
265                        }
266                        for (Map<String, Object> map : processed.values()) {
267                            mergedList.add(map);
268                        }
269                    } else {
270                        // if both are lists, simply combine them, first the defaults, then the content
271                        // just make sure not to add the same value twice
272                        mergedList.addAll(defaultList);
273                        for (Object o : contentList) {
274                            if (!mergedList.contains(o)) {
275                                mergedList.add(o);
276                            }
277                        }
278                    }
279                    content.put(defaultEntry.getKey(), mergedList);
280                }
281            }
282        }
283    }

285    private static boolean allListValuesAreMapsOfOne(List list) {
286        for (Object o : list) {
287            if (!(o instanceof Map)) {
288                return false;
289            }
290            if (((Map) o).size() != 1) {
291                return false;
292            }
293        }
294        return true;
295    }

297    public static void copyCurrentStructure(XContentGenerator generator, XContentParser parser) throws IOException {
298        XContentParser.Token t = parser.currentToken();
299
300        // Let's handle field-name separately first
301        if (t == XContentParser.Token.FIELD_NAME) {
302            generator.writeFieldName(parser.currentName());
303            t = parser.nextToken();
304            // fall-through to copy the associated value
305        }
306
307        switch (t) {
308            case START_ARRAY:
309                generator.writeStartArray();
310                while (parser.nextToken() != XContentParser.Token.END_ARRAY) {
311                    copyCurrentStructure(generator, parser);
312                }
313                generator.writeEndArray();
314                break;
315            case START_OBJECT:
316                generator.writeStartObject();
317                while (parser.nextToken() != XContentParser.Token.END_OBJECT) {
318                    copyCurrentStructure(generator, parser);
319                }
320                generator.writeEndObject();
321                break;
322            default: // others are simple:
323                copyCurrentEvent(generator, parser);
324        }
325    }

327    public static void copyCurrentEvent(XContentGenerator generator, XContentParser parser) throws IOException {
328        switch (parser.currentToken()) {
329            case START_OBJECT:
330                generator.writeStartObject();
331                break;
332            case END_OBJECT:
333                generator.writeEndObject();
334                break;
335            case START_ARRAY:
336                generator.writeStartArray();
337                break;
338            case END_ARRAY:
339                generator.writeEndArray();
340                break;
341            case FIELD_NAME:
342                generator.writeFieldName(parser.currentName());
343                break;
344            case VALUE_STRING:
345                if (parser.hasTextCharacters()) {
346                    generator.writeString(parser.textCharacters(), parser.textOffset(), parser.textLength());
347                } else {
348                    generator.writeString(parser.text());
349                }
350                break;
351            case VALUE_NUMBER:
352                switch (parser.numberType()) {
353                    case INT:
354                        generator.writeNumber(parser.intValue());
355                        break;
356                    case LONG:
357                        generator.writeNumber(parser.longValue());
358                        break;
359                    case FLOAT:
360                        generator.writeNumber(parser.floatValue());
361                        break;
362                    case DOUBLE:
363                        generator.writeNumber(parser.doubleValue());
364                        break;
365                }
366                break;
367            case VALUE_BOOLEAN:
368                generator.writeBoolean(parser.booleanValue());
369                break;
370            case VALUE_NULL:
371                generator.writeNull();
372                break;
373            case VALUE_EMBEDDED_OBJECT:
374                generator.writeBinary(parser.binaryValue());
375        }
