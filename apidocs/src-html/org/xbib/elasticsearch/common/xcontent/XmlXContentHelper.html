029        public static XContentParser createParser(BytesReference bytes) throws IOException {
030            if (bytes.hasArray()) {
031                return createParser(bytes.array(), bytes.arrayOffset(), bytes.length());
032            }
033            Compressor compressor = CompressorFactory.compressor(bytes);
034            if (compressor != null) {
035                CompressedStreamInput compressedInput = compressor.streamInput(bytes.streamInput());
036                XmlXContentType contentType = XmlXContentFactory.xContentType(compressedInput);
037                compressedInput.resetToBufferStart();
038                return XmlXContentFactory.xContent(contentType).createParser(compressedInput);
039            } else {
040                return XmlXContentFactory.xContent(bytes).createParser(bytes.streamInput());
041            }
042        }
043    
044        public static XContentParser createParser(byte[] data, int offset, int length) throws IOException {
045            Compressor compressor = CompressorFactory.compressor(data, offset, length);
046            if (compressor != null) {
047                CompressedStreamInput compressedInput = compressor.streamInput(new BytesStreamInput(data, offset, length, false));
048                XmlXContentType contentType = XmlXContentFactory.xContentType(compressedInput);
049                compressedInput.resetToBufferStart();
050                return XmlXContentFactory.xContent(contentType).createParser(compressedInput);
051            } else {
052                return XmlXContentFactory.xContent(data, offset, length).createParser(data, offset, length);
053            }
054        }

056        public static Tuple<XmlXContentType, Map<String, Object>> convertToMap(BytesReference bytes, boolean ordered)
057                throws ElasticsearchParseException {
058            if (bytes.hasArray()) {
059                return convertToMap(bytes.array(), bytes.arrayOffset(), bytes.length(), ordered);
060            }
061            try {
062                XContentParser parser;
063                XmlXContentType contentType;
064                Compressor compressor = CompressorFactory.compressor(bytes);
065                if (compressor != null) {
066                    CompressedStreamInput compressedStreamInput = compressor.streamInput(bytes.streamInput());
067                    contentType = XmlXContentFactory.xContentType(compressedStreamInput);
068                    compressedStreamInput.resetToBufferStart();
069                    parser = XmlXContentFactory.xContent(contentType).createParser(compressedStreamInput);
070                } else {
071                    contentType = XmlXContentFactory.xContentType(bytes);
072                    parser = XmlXContentFactory.xContent(contentType).createParser(bytes.streamInput());
073                }
074                if (ordered) {
075                    return Tuple.tuple(contentType, parser.mapOrderedAndClose());
076                } else {
077                    return Tuple.tuple(contentType, parser.mapAndClose());
078                }
079            } catch (IOException e) {
080                throw new ElasticsearchParseException("Failed to parse content to map", e);
081            }
082        }

118        public static String convertToJson(BytesReference bytes, boolean reformatJson, boolean prettyPrint) throws IOException {
119            if (bytes.hasArray()) {
120                return convertToJson(bytes.array(), bytes.arrayOffset(), bytes.length(), reformatJson, prettyPrint);
121            }
122            XmlXContentType xmlXContentType = XmlXContentFactory.xContentType(bytes);
123            if (xmlXContentType == XmlXContentType.JSON && !reformatJson) {
124                BytesArray bytesArray = bytes.toBytesArray();
125                return new String(bytesArray.array(), bytesArray.arrayOffset(), bytesArray.length(), Charsets.UTF_8);
126            }
127            XContentParser parser = null;
128            try {
129                parser = XmlXContentFactory.xContent(xmlXContentType).createParser(bytes.streamInput());
130                parser.nextToken();
131                XmlXContentBuilder builder = XmlXContentFactory.jsonBuilder();
132                if (prettyPrint) {
133                    builder.prettyPrint();
134                }
135                builder.copyCurrentStructure(parser);
136                return builder.string();
137            } finally {
138                if (parser != null) {
139                    parser.close();
140                }
141            }
142        }

193        /**
194         * Updates the provided changes into the source. If the key exists in the changes, it overrides the one in source
195         * unless both are Maps, in which case it recuersively updated it.
196         */
197        public static void update(Map<String, Object> source, Map<String, Object> changes) {
198            for (Map.Entry<String, Object> changesEntry : changes.entrySet()) {
199                if (!source.containsKey(changesEntry.getKey())) {
200                    // safe to copy, change does not exist in source
201                    source.put(changesEntry.getKey(), changesEntry.getValue());
202                } else {
203                    if (source.get(changesEntry.getKey()) instanceof Map && changesEntry.getValue() instanceof Map) {
204                        // recursive merge maps
205                        update((Map<String, Object>) source.get(changesEntry.getKey()), (Map<String, Object>) changesEntry.getValue());
206                    } else {
207                        // update the field
208                        source.put(changesEntry.getKey(), changesEntry.getValue());
209                    }
210                }
211            }
212        }

214        /**
215         * Merges the defaults provided as the second parameter into the content of the first. Only does recursive merge
216         * for inner maps.
217         */
218        @SuppressWarnings({"unchecked"})
219        public static void mergeDefaults(Map<String, Object> content, Map<String, Object> defaults) {
220            for (Map.Entry<String, Object> defaultEntry : defaults.entrySet()) {
221                if (!content.containsKey(defaultEntry.getKey())) {
222                    // copy it over, it does not exists in the content
223                    content.put(defaultEntry.getKey(), defaultEntry.getValue());
224                } else {
225                    // in the content and in the default, only merge compound ones (maps)
226                    if (content.get(defaultEntry.getKey()) instanceof Map && defaultEntry.getValue() instanceof Map) {
227                        mergeDefaults((Map<String, Object>) content.get(defaultEntry.getKey()), (Map<String, Object>) defaultEntry.getValue());
228                    } else if (content.get(defaultEntry.getKey()) instanceof List && defaultEntry.getValue() instanceof List) {
229                        List defaultList = (List) defaultEntry.getValue();
230                        List contentList = (List) content.get(defaultEntry.getKey());
231    
232                        List mergedList = new ArrayList();
233                        if (allListValuesAreMapsOfOne(defaultList) && allListValuesAreMapsOfOne(contentList)) {
234                            // all are in the form of [ {"key1" : {}}, {"key2" : {}} ], merge based on keys
235                            Map<String, Map<String, Object>> processed = Maps.newLinkedHashMap();
236                            for (Object o : contentList) {
237                                Map<String, Object> map = (Map<String, Object>) o;
238                                Map.Entry<String, Object> entry = map.entrySet().iterator().next();
239                                processed.put(entry.getKey(), map);
240                            }
241                            for (Object o : defaultList) {
242                                Map<String, Object> map = (Map<String, Object>) o;
243                                Map.Entry<String, Object> entry = map.entrySet().iterator().next();
244                                if (processed.containsKey(entry.getKey())) {
245                                    mergeDefaults(processed.get(entry.getKey()), map);
246                                }
247                            }
248                            for (Map<String, Object> map : processed.values()) {
249                                mergedList.add(map);
250                            }
251                        } else {
252                            // if both are lists, simply combine them, first the defaults, then the content
253                            // just make sure not to add the same value twice
254                            mergedList.addAll(defaultList);
255                            for (Object o : contentList) {
256                                if (!mergedList.contains(o)) {
257                                    mergedList.add(o);
258                                }
259                            }
260                        }
261                        content.put(defaultEntry.getKey(), mergedList);
262                    }
263                }
264            }
265        }

267        private static boolean allListValuesAreMapsOfOne(List list) {
268            for (Object o : list) {
269                if (!(o instanceof Map)) {
270                    return false;
271                }
272                if (((Map) o).size() != 1) {
273                    return false;
274                }
275            }
276            return true;
277        }

279        public static void copyCurrentStructure(XContentGenerator generator, XContentParser parser) throws IOException {
280            XContentParser.Token t = parser.currentToken();
281    
282            // Let's handle field-name separately first
283            if (t == XContentParser.Token.FIELD_NAME) {
284                generator.writeFieldName(parser.currentName());
285                t = parser.nextToken();
286                // fall-through to copy the associated value
287            }
288    
289            switch (t) {
290                case START_ARRAY:
291                    generator.writeStartArray();
292                    while (parser.nextToken() != XContentParser.Token.END_ARRAY) {
293                        copyCurrentStructure(generator, parser);
294                    }
295                    generator.writeEndArray();
296                    break;
297                case START_OBJECT:
298                    generator.writeStartObject();
299                    while (parser.nextToken() != XContentParser.Token.END_OBJECT) {
300                        copyCurrentStructure(generator, parser);
301                    }
302                    generator.writeEndObject();
303                    break;
304                default: // others are simple:
305                    copyCurrentEvent(generator, parser);
306            }
307        }

309        public static void copyCurrentEvent(XContentGenerator generator, XContentParser parser) throws IOException {
310            switch (parser.currentToken()) {
311                case START_OBJECT:
312                    generator.writeStartObject();
313                    break;
314                case END_OBJECT:
315                    generator.writeEndObject();
316                    break;
317                case START_ARRAY:
318                    generator.writeStartArray();
319                    break;
320                case END_ARRAY:
321                    generator.writeEndArray();
322                    break;
323                case FIELD_NAME:
324                    generator.writeFieldName(parser.currentName());
325                    break;
326                case VALUE_STRING:
327                    if (parser.hasTextCharacters()) {
328                        generator.writeString(parser.textCharacters(), parser.textOffset(), parser.textLength());
329                    } else {
330                        generator.writeString(parser.text());
331                    }
332                    break;
333                case VALUE_NUMBER:
334                    switch (parser.numberType()) {
335                        case INT:
336                            generator.writeNumber(parser.intValue());
337                            break;
338                        case LONG:
339                            generator.writeNumber(parser.longValue());
340                            break;
341                        case FLOAT:
342                            generator.writeNumber(parser.floatValue());
343                            break;
344                        case DOUBLE:
345                            generator.writeNumber(parser.doubleValue());
346                            break;
347                    }
348                    break;
349                case VALUE_BOOLEAN:
350                    generator.writeBoolean(parser.booleanValue());
351                    break;
352                case VALUE_NULL:
353                    generator.writeNull();
354                    break;
355                case VALUE_EMBEDDED_OBJECT:
356                    generator.writeBinary(parser.binaryValue());
357            }
358        }
