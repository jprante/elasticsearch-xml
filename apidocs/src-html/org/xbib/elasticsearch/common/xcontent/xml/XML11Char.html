024/**
025 * This class defines the basic properties of characters in XML 1.1. The data
026 * in this class can be used to verify that a character is a valid
027 * XML 1.1 character or if the character is a space, name start, or name
028 * character.
029 * <p>
030 * A series of convenience methods are supplied to ease the burden
031 * of the developer.  Using the character as an index into the <code>XML11CHARS</code>
032 * array and applying the appropriate mask flag (e.g.
033 * <code>MASK_VALID</code>), yields the same results as calling the
034 * convenience methods. There is one exception: check the comments
035 * for the <code>isValid</code> method for details.
036 *
037 * Taken from org.apache.xerces.util.XML11Char.java
038 *
039 */
040public class XML11Char {

132    /**
133     * Returns true if the specified character is a space character
134     * as amdended in the XML 1.1 specification.
135     *
136     * @param c The character to check.
137     */
138    public static boolean isXML11Space(int c) {
139        return (c < 0x10000 && (XML11CHARS[c] & MASK_XML11_SPACE) != 0);
140    }
141
142    /**
143     * Returns true if the specified character is valid. This method
144     * also checks the surrogate character range from 0x10000 to 0x10FFFF.
145     * <p>
146     * If the program chooses to apply the mask directly to the
147     * <code>XML11CHARS</code> array, then they are responsible for checking
148     * the surrogate character range.
149     *
150     * @param c The character to check.
151     */
152    public static boolean isXML11Valid(int c) {
153        return (c < 0x10000 && (XML11CHARS[c] & MASK_XML11_VALID) != 0) 
154                || (0x10000 <= c && c <= 0x10FFFF);
155    }
156
157    /**
158     * Returns true if the specified character is invalid.
159     *
160     * @param c The character to check.
161     */
162    public static boolean isXML11Invalid(int c) {
163        return !isXML11Valid(c);
164    }
165
166    /**
167     * Returns true if the specified character is valid and permitted outside
168     * of a character reference.  
169     * That is, this method will return false for the same set as
170     * isXML11Valid, except it also reports false for "control characters".
171     *
172     * @param c The character to check.
173     */
174    public static boolean isXML11ValidLiteral(int c) {
175        return ((c < 0x10000 && ((XML11CHARS[c] & MASK_XML11_VALID) != 0 && (XML11CHARS[c] & MASK_XML11_CONTROL) == 0))
176            || (0x10000 <= c && c <= 0x10FFFF)); 
177    }

249    /**
250     * Returns whether the given character is a valid 
251     * high surrogate for a name character. This includes
252     * all high surrogates for characters [0x10000-0xEFFFF].
253     * In other words everything excluding planes 15 and 16.
254     *
255     * @param c The character to check.
256     */
257    public static boolean isXML11NameHighSurrogate(int c) {
258        return (0xD800 <= c && c <= 0xDB7F);
259    }

261    /**
262     * Check to see if a string is a valid Name
263     *
264     * @param name string to check
265     * @return true if name is a valid Name
266     */
267    public static boolean isXML11ValidName(String name) {
268        int length = name.length();
269        if (length == 0)
270            return false;
271        int i = 1;
272        char ch = name.charAt(0);
273        if( !isXML11NameStart(ch) ) {
274            if ( length > 1 && isXML11NameHighSurrogate(ch) ) {
275                char ch2 = name.charAt(1);
276                if ( !XMLChar.isLowSurrogate(ch2) || 
277                     !isXML11NameStart(XMLChar.supplemental(ch, ch2)) ) {
278                    return false;
279                }
280                i = 2;
281            }
282            else {
283                return false;
284            }
285        }
286        while (i < length) {
287            ch = name.charAt(i);
288            if ( !isXML11Name(ch) ) {
289                if ( ++i < length && isXML11NameHighSurrogate(ch) ) {
290                    char ch2 = name.charAt(i);
291                    if ( !XMLChar.isLowSurrogate(ch2) || 
292                         !isXML11Name(XMLChar.supplemental(ch, ch2)) ) {
293                        return false;
294                    }
295                }
296                else {
297                    return false;
298                }
299            }
300            ++i;
301        }
302        return true;
303    }

308     *
309     * @param ncName string to check

323                    return false;
324                }

328                return false;
329            }
330        }

333            if ( !isXML11NCName(ch) ) {
334                if ( ++i < length && isXML11NameHighSurrogate(ch) ) {

346        }

352     * Check to see if a string is a valid Nmtoken
353     *

357    public static boolean isXML11ValidNmtoken(String nmtoken) {
358        int length = nmtoken.length();

362            char ch = nmtoken.charAt(i);
363            if( !isXML11Name(ch) ) {
364                if ( ++i < length && isXML11NameHighSurrogate(ch) ) {
365                    char ch2 = nmtoken.charAt(i);
366                    if ( !XMLChar.isLowSurrogate(ch2) || 
367                         !isXML11Name(XMLChar.supplemental(ch, ch2)) ) {
368                        return false;
369                    }
370                }
371                else {
372                    return false;
373                }
374            }
375        }
376        return true;
377    }
