103    @Override
104    public void writeStartObject() throws IOException {
105        try {
106            if (!started) {
107                generator.getStaxWriter().setDefaultNamespace(params.getQName().getNamespaceURI());
108                generator.startWrappedValue(null, params.getQName());
109            }
110            generator.writeStartObject();
111            if (!started) {
112                for (String prefix : getNamespaceContext().getNamespaces().keySet()) {
113                    generator.getStaxWriter().writeNamespace(prefix, getNamespaceContext().getNamespaceURI(prefix));
114                }
115                started = true;
116            }
117        } catch (XMLStreamException e) {
118            throw new IOException(e);
119        }
120    }

140        generator.writeString(text);
141        if (context && prefix != null) {
142            try {
143                params.getNamespaceContext().addNamespace(prefix, text);
144                generator.getStaxWriter().writeNamespace(prefix, text);
145            } catch (XMLStreamException e) {
146                throw new IOException(e);
147            }
148            prefix = null;
149        }

367            parser = XmlXContent.xmlFactory().createParser(content.array(), content.arrayOffset(), content.length());
368        } else {
369            parser = XmlXContent.xmlFactory().createParser(content.streamInput());
370        }
371        try {
372            parser.nextToken();
373            generator.copyCurrentStructure(parser);
374        } finally {
375            parser.close();
376        }

391    @Override
392    public void copyCurrentStructure(XContentParser parser) throws IOException {
393        if (parser.currentToken() == null) {
394            parser.nextToken();
395        }
396        if (parser instanceof XmlXContentParser) {
397            generator.copyCurrentStructure(((XmlXContentParser) parser).parser);
398        } else {
399            XmlXContentHelper.copyCurrentStructure(this, parser);
400        }
401    }

413    private void writeFieldNameXml(String name) throws IOException {
414        if (!context) {
415            this.context = "@context".equals(name);
416            this.prefix = null;
417        }
418        if (name.startsWith("@")) {
419            // setting to attribute is simple but tricky, it allows to declare namespaces in StaX
420            generator.setNextIsAttribute(true);
421        } else if (context) {
422            prefix = name;
423        }
424        QName qname = toQName(name);
425        generator.setNextName(qname);
426        generator.writeFieldName(qname.getLocalPart());
427    }
428
429    private QName toQName(String name) throws IOException {
430        QName root = params.getQName();
431        XmlNamespaceContext context = params.getNamespaceContext();
432        String nsPrefix = root.getPrefix();
433        String nsURI = root.getNamespaceURI();
434        if (name.startsWith("_") || name.startsWith("@")) {
435            name = name.substring(1);
436        }
437        name = ISO9075.encode(name);
438        int pos = name.indexOf(':');
439        if (pos > 0) {
440            nsPrefix = name.substring(0, pos);
441            nsURI = context.getNamespaceURI(nsPrefix);
442            if (nsURI == null) {
443                throw new IOException("unknown namespace prefix: " + nsPrefix);
444            }
445            name = name.substring(pos + 1);
446        }
447        return new QName(nsURI, name, nsPrefix);
448    }
