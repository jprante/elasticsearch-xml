080        @Override
081        public void writeStartObject() throws IOException {
082            try {
083                if (!started) {
084                    generator.getStaxWriter().setDefaultNamespace(params.getQName().getNamespaceURI());
085                    generator.startWrappedValue(null, params.getQName());
086                }
087                generator.writeStartObject();
088                if (!started ) {
089                    if (getNamespaceContext() != null &&  getNamespaceContext().getNamespaces() != null) {
090                        for (String prefix : getNamespaceContext().getNamespaces().keySet()) {
091                            generator.getStaxWriter().writeNamespace(prefix, getNamespaceContext().getNamespaceURI(prefix));
092                        }
093                    }
094                    started = true;
095                }
096            } catch (XMLStreamException e) {
097                throw new IOException(e);
098            }
099        }

119            generator.writeString(text);
120            if (context && prefix != null) {
121                try {
122                    params.getNamespaceContext().addNamespace(prefix, text);
123                    generator.getStaxWriter().writeNamespace(prefix, text);
124                } catch (XMLStreamException e) {
125                    throw new IOException(e);
126                }
127                prefix = null;
128            }
129        }

319            writeFieldNameXml(fieldName);
320            JsonParser parser = XmlXContent.xmlFactory().createParser(content);
321            try {
322                parser.nextToken();
323                generator.copyCurrentStructure(parser);
324            } finally {
325                parser.close();
326            }

343            writeFieldNameXml(fieldName);
344            JsonParser parser;
345            if (content.hasArray()) {
346                parser = XmlXContent.xmlFactory().createParser(content.array(), content.arrayOffset(), content.length());
347            } else {
348                parser = XmlXContent.xmlFactory().createParser(content.streamInput());
349            }
350            try {
351                parser.nextToken();
352                generator.copyCurrentStructure(parser);
353            } finally {
354                parser.close();
355            }

370        @Override
371        public void copyCurrentStructure(XContentParser parser) throws IOException {
372            if (parser.currentToken() == null) {
373                parser.nextToken();
374            }
375            if (parser instanceof XmlXContentParser) {
376                generator.copyCurrentStructure(((XmlXContentParser) parser).parser);
377            } else {
378                XmlXContentHelper.copyCurrentStructure(this, parser);
379            }
380        }

392        private void writeFieldNameXml(String name) throws IOException {
393            if (!context) {
394                this.context = "@context".equals(name);
395                this.prefix = null;
396            }
397            if (name.startsWith("@")) {
398                // setting to attribute is simple but tricky, it allows to declare namespaces in StaX
399                generator.setNextIsAttribute(true);
400            } else if (context) {
401                prefix = name;
402            }
403            QName qname = toQName(name);
404            generator.setNextName(qname);
405            generator.writeFieldName(qname.getLocalPart());
406        }
407    
408        private QName toQName(String name) throws IOException {
409            QName root = params.getQName();
410            XmlNamespaceContext context = params.getNamespaceContext();
411            String nsPrefix = root.getPrefix();
412            String nsURI = root.getNamespaceURI();
413            if (name.startsWith("_") || name.startsWith("@")) {
414                name = name.substring(1);
415            }
416            name = ISO9075.encode(name);
417            int pos = name.indexOf(':');
418            if (pos > 0) {
419                nsPrefix = name.substring(0, pos);
420                nsURI = context != null? context.getNamespaceURI(nsPrefix) : XmlXParams.DEFAULT_ROOT.getNamespaceURI();
421                if (nsURI == null) {
422                    throw new IOException("unknown namespace prefix: " + nsPrefix);
423                }
424                name = name.substring(pos + 1);
425            }
426            return new QName(nsURI, name, nsPrefix);
427        }
