002    package org.xbib.elasticsearch.common.xcontent.xml;
003    
004    import org.elasticsearch.common.collect.Maps;
005    
006    import javax.xml.namespace.NamespaceContext;
007    import java.util.Enumeration;
008    import java.util.HashSet;
009    import java.util.Iterator;
010    import java.util.Map;
011    import java.util.MissingResourceException;
012    import java.util.ResourceBundle;
013    import java.util.Set;
014    import java.util.SortedMap;
015    
016    /**
017     * Contains a simple context for XML namespaces
018     *
019     */
020    public class XmlNamespaceContext implements NamespaceContext {
021    
022        private static final String DEFAULT_RESOURCE = "xml-namespaces";
023    
024        private final SortedMap<String, String> namespaces = Maps.newTreeMap();
025    
026        private final SortedMap<String, Set<String>> prefixes = Maps.newTreeMap();
027    
028        protected XmlNamespaceContext() {
029        }
030    
031        protected XmlNamespaceContext(ResourceBundle bundle) {
032            Enumeration<String> en = bundle.getKeys();
033            while (en.hasMoreElements()) {
034                String prefix = en.nextElement();
035                String namespace = bundle.getString(prefix);
036                addNamespace(prefix, namespace);
037            }
038        }
039    
040        protected static String bundleName() {
041            return DEFAULT_RESOURCE;
042        }
043    
044        /**
045         * Empty namespace context.
046         *
047         * @return a namespace context instance
048         */
049        public static XmlNamespaceContext newInstance() {
050            return new XmlNamespaceContext();
051        }
052    
053        /**
054         * Default namespaces from default resource bundle.
055         * @return a namespace context instance
056         */
057        public static XmlNamespaceContext getDefaultInstance() {
058            return newInstance(bundleName());
059        }
060    
061        /**
062         * Namespaces from given resource bundle name.
063         * @return a namespace context instance
064         */
065        public static XmlNamespaceContext newInstance(String bundleName) {
066            try {
067                return new XmlNamespaceContext(ResourceBundle.getBundle(bundleName));
068            } catch (MissingResourceException e) {
069                return new XmlNamespaceContext();
070            }
071        }
072    
073        public final synchronized void addNamespace(String prefix, String namespace) {
074            namespaces.put(prefix, namespace);
075            if (prefixes.containsKey(namespace)) {
076                prefixes.get(namespace).add(prefix);
077            } else {
078                Set<String> set = new HashSet<String>();
079                set.add(prefix);
080                prefixes.put(namespace, set);
081            }
082        }
083    
084        public Map<String, String> getNamespaces() {
085            return namespaces;
086        }
087    
088        @Override
089        public String getNamespaceURI(String prefix) {
090            if (prefix == null) {
091                return null;
092            }
093            return namespaces.containsKey(prefix) ? namespaces.get(prefix) : null;
094        }
095    
096        @Override
097        public String getPrefix(String namespaceURI) {
098            Iterator<String> it = getPrefixes(namespaceURI);
099            return it != null && it.hasNext() ? it.next() : null;
100        }
101    
102        @Override
103        public Iterator<String> getPrefixes(String namespace) {
104            if (namespace == null) {
105                throw new IllegalArgumentException("namespace URI cannot be null");
106            }
107            return prefixes.containsKey(namespace) ?
108                    prefixes.get(namespace).iterator() : null;
109        }
110    
111        public String toString() {
112            return namespaces.toString();
113        }
114    
115    }
