020package org.xbib.elasticsearch.common.xcontent.xml;
021
022import com.fasterxml.jackson.core.JsonEncoding;
023import com.fasterxml.jackson.dataformat.xml.XmlFactory;
024import org.elasticsearch.common.bytes.BytesReference;
025import org.elasticsearch.common.io.FastStringReader;
026import org.elasticsearch.common.xcontent.XContent;
027import org.elasticsearch.common.xcontent.XContentType;
028import org.xbib.elasticsearch.common.xcontent.XmlXContentBuilder;
029import org.elasticsearch.common.xcontent.XContentGenerator;
030import org.elasticsearch.common.xcontent.XContentParser;
031
032import javax.xml.stream.XMLInputFactory;
033import javax.xml.stream.XMLOutputFactory;
034import java.io.IOException;
035import java.io.InputStream;
036import java.io.OutputStream;
037import java.io.Reader;
038import java.io.Writer;
039
040/**
041 * A XML based content implementation using Jackson XML dataformat
042 *
043 */
044public class XmlXContent implements XContent {
045
046    private final static XMLInputFactory inputFactory = XMLInputFactory.newInstance();
047
048    private final static XMLOutputFactory outputFactory = XMLOutputFactory.newInstance();
049
050    static {
051        inputFactory.setProperty("javax.xml.stream.isNamespaceAware", Boolean.TRUE);
052        inputFactory.setProperty("javax.xml.stream.isValidating", Boolean.FALSE);
053        inputFactory.setProperty("javax.xml.stream.isCoalescing", Boolean.TRUE);
054        inputFactory.setProperty("javax.xml.stream.isReplacingEntityReferences", Boolean.FALSE);
055        inputFactory.setProperty("javax.xml.stream.isSupportingExternalEntities", Boolean.FALSE);
056
057        outputFactory.setProperty("javax.xml.stream.isRepairingNamespaces", Boolean.TRUE);
058
059        xmlFactory = new XmlFactory(inputFactory, outputFactory);
060
061        xmlXContent = new XmlXContent();
062    }
063
064    public static XmlXContentBuilder contentBuilder() throws IOException {
065        return XmlXContentBuilder.builder(xmlXContent);
066    }
067
068    public static XmlXContentBuilder contentBuilder(XmlXParams params) throws IOException {
069        XmlXContentBuilder builder = XmlXContentBuilder.builder(xmlXContent);
070        ((XmlXContentGenerator) builder.generator()).setParams(params);
071        return builder;
072    }
073
074    private final static XmlFactory xmlFactory;
075
076    private final static XmlXContent xmlXContent;
077
078    private XmlXContent() {
079    }
080
081    @Override
082    public XContentType type() {
083        //return XmlXContentType.XML;
084        return null;
085    }
086
087    public static XmlXContent xmlXContent() {
088        return xmlXContent;
089    }
090
091    protected static XmlFactory xmlFactory() {
092        return xmlFactory;
093    }
094
095    @Override
096    public byte streamSeparator() {
097        throw new UnsupportedOperationException("xml does not support stream parsing...");
098    }
099
100    @Override
101    public XContentGenerator createGenerator(OutputStream os) throws IOException {
102        return new XmlXContentGenerator(xmlFactory.createGenerator(os, JsonEncoding.UTF8));
103    }
104
105    @Override
106    public XContentGenerator createGenerator(Writer writer) throws IOException {
107        return new XmlXContentGenerator(xmlFactory.createGenerator(writer));
108    }
109
110    @Override
111    public XContentParser createParser(String content) throws IOException {
112        return new XmlXContentParser(xmlFactory.createParser(new FastStringReader(content)));
113    }
114
115    @Override
116    public XContentParser createParser(InputStream is) throws IOException {
117        return new XmlXContentParser(xmlFactory.createParser(is));
118    }
119
120    @Override
121    public XContentParser createParser(byte[] data) throws IOException {
122        return new XmlXContentParser(xmlFactory.createParser(data));
123    }
124
125    @Override
126    public XContentParser createParser(byte[] data, int offset, int length) throws IOException {
127        return new XmlXContentParser(xmlFactory.createParser(data, offset, length));
128    }
129
130    @Override
131    public XContentParser createParser(BytesReference bytes) throws IOException {
132        if (bytes.hasArray()) {
133            return createParser(bytes.array(), bytes.arrayOffset(), bytes.length());
134        }
135        return createParser(bytes.streamInput());
136    }
137
138    @Override
139    public XContentParser createParser(Reader reader) throws IOException {
140        return new XmlXContentParser(xmlFactory.createParser(reader));
141    }
142}
