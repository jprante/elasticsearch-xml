001
002package org.xbib.elasticsearch.common.xcontent.xml;
003
004import com.fasterxml.jackson.core.JsonEncoding;
005import com.fasterxml.jackson.dataformat.xml.XmlFactory;
006import org.elasticsearch.common.bytes.BytesReference;
007import org.elasticsearch.common.io.FastStringReader;
008import org.elasticsearch.common.xcontent.XContent;
009import org.elasticsearch.common.xcontent.XContentType;
010import org.xbib.elasticsearch.common.xcontent.XmlXContentBuilder;
011import org.elasticsearch.common.xcontent.XContentGenerator;
012import org.elasticsearch.common.xcontent.XContentParser;
013
014import javax.xml.stream.XMLInputFactory;
015import javax.xml.stream.XMLOutputFactory;
016import java.io.IOException;
017import java.io.InputStream;
018import java.io.OutputStream;
019import java.io.Reader;
020import java.io.Writer;
021
022/**
023 * A XML based content implementation using Jackson XML dataformat
024 *
025 */
026public class XmlXContent implements XContent {
027
028    private final static XMLInputFactory inputFactory = XMLInputFactory.newInstance();
029
030    private final static XMLOutputFactory outputFactory = XMLOutputFactory.newInstance();
031
032    static {
033        inputFactory.setProperty("javax.xml.stream.isNamespaceAware", Boolean.TRUE);
034        inputFactory.setProperty("javax.xml.stream.isValidating", Boolean.FALSE);
035        inputFactory.setProperty("javax.xml.stream.isCoalescing", Boolean.TRUE);
036        inputFactory.setProperty("javax.xml.stream.isReplacingEntityReferences", Boolean.FALSE);
037        inputFactory.setProperty("javax.xml.stream.isSupportingExternalEntities", Boolean.FALSE);
038
039        outputFactory.setProperty("javax.xml.stream.isRepairingNamespaces", Boolean.TRUE);
040
041        xmlFactory = new XmlFactory(inputFactory, outputFactory);
042
043        xmlXContent = new XmlXContent();
044    }
045
046    public static XmlXContentBuilder contentBuilder() throws IOException {
047        return XmlXContentBuilder.builder(xmlXContent);
048    }
049
050    public static XmlXContentBuilder contentBuilder(XmlXParams params) throws IOException {
051        XmlXContentBuilder builder = XmlXContentBuilder.builder(xmlXContent);
052        ((XmlXContentGenerator) builder.generator()).setParams(params);
053        return builder;
054    }
055
056    private final static XmlFactory xmlFactory;
057
058    private final static XmlXContent xmlXContent;
059
060    private XmlXContent() {
061    }
062
063    @Override
064    public XContentType type() {
065        //return XmlXContentType.XML;
066        return null;
067    }
068
069    public static XmlXContent xmlXContent() {
070        return xmlXContent;
071    }
072
073    protected static XmlFactory xmlFactory() {
074        return xmlFactory;
075    }
076
077    @Override
078    public byte streamSeparator() {
079        throw new UnsupportedOperationException("xml does not support stream parsing...");
080    }
081
082    @Override
083    public XContentGenerator createGenerator(OutputStream os) throws IOException {
084        return new XmlXContentGenerator(xmlFactory.createGenerator(os, JsonEncoding.UTF8));
085    }
086
087    @Override
088    public XContentGenerator createGenerator(Writer writer) throws IOException {
089        return new XmlXContentGenerator(xmlFactory.createGenerator(writer));
090    }
091
092    @Override
093    public XContentParser createParser(String content) throws IOException {
094        return new XmlXContentParser(xmlFactory.createParser(new FastStringReader(content)));
095    }
096
097    @Override
098    public XContentParser createParser(InputStream is) throws IOException {
099        return new XmlXContentParser(xmlFactory.createParser(is));
100    }
101
102    @Override
103    public XContentParser createParser(byte[] data) throws IOException {
104        return new XmlXContentParser(xmlFactory.createParser(data));
105    }
106
107    @Override
108    public XContentParser createParser(byte[] data, int offset, int length) throws IOException {
109        return new XmlXContentParser(xmlFactory.createParser(data, offset, length));
110    }
111
112    @Override
113    public XContentParser createParser(BytesReference bytes) throws IOException {
114        if (bytes.hasArray()) {
115            return createParser(bytes.array(), bytes.arrayOffset(), bytes.length());
116        }
117        return createParser(bytes.streamInput());
118    }
119
120    @Override
121    public XContentParser createParser(Reader reader) throws IOException {
122        return new XmlXContentParser(xmlFactory.createParser(reader));
123    }
124}
