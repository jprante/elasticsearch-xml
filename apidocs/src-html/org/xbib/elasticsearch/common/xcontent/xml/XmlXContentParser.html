002    package org.xbib.elasticsearch.common.xcontent.xml;
003    
004    import com.fasterxml.jackson.core.JsonParser;
005    import com.fasterxml.jackson.core.JsonToken;
006    import org.apache.lucene.util.BytesRef;
007    import org.apache.lucene.util.UnicodeUtil;
008    import org.elasticsearch.common.xcontent.XContentParser;
009    import org.elasticsearch.common.xcontent.XContentType;
010    import org.elasticsearch.common.xcontent.support.AbstractXContentParser;
011    
012    import java.io.IOException;
013    
014    /**
015     *
016     */
017    public class XmlXContentParser extends AbstractXContentParser {
018    
019        final JsonParser parser;
020    
021        public XmlXContentParser(JsonParser parser) {
022            this.parser = parser;
023        }
024    
025    
026        @Override
027        public XContentType contentType() {
028            //return XmlXContentType.XML;
029            return null;
030        }
031    
032    
033        @Override
034        public XContentParser.Token nextToken() throws IOException {
035            return convertToken(parser.nextToken());
036        }
037    
038    
039        @Override
040        public void skipChildren() throws IOException {
041            parser.skipChildren();
042        }
043    
044    
045        @Override
046        public XContentParser.Token currentToken() {
047            return convertToken(parser.getCurrentToken());
048        }
049    
050    
051        @Override
052        public XContentParser.NumberType numberType() throws IOException {
053            return convertNumberType(parser.getNumberType());
054        }
055    
056    
057        @Override
058        public boolean estimatedNumberType() {
059            return true;
060        }
061    
062    
063        @Override
064        public String currentName() throws IOException {
065            return parser.getCurrentName();
066        }
067    
068    
069        @Override
070        protected boolean doBooleanValue() throws IOException {
071            return parser.getBooleanValue();
072        }
073    
074    
075        @Override
076        public String text() throws IOException {
077            return parser.getText();
078        }
079    
080        @Override
081        public BytesRef bytes() throws IOException {
082            BytesRef bytes = new BytesRef();
083            UnicodeUtil.UTF16toUTF8(parser.getTextCharacters(), parser.getTextOffset(), parser.getTextLength(), bytes);
084            return bytes;
085        }
086    
087        @Override
088        public Object objectText() throws IOException {
089            JsonToken currentToken = parser.getCurrentToken();
090            if (currentToken == JsonToken.VALUE_STRING) {
091                return text();
092            } else if (currentToken == JsonToken.VALUE_NUMBER_INT || currentToken == JsonToken.VALUE_NUMBER_FLOAT) {
093                return parser.getNumberValue();
094            } else if (currentToken == JsonToken.VALUE_TRUE) {
095                return Boolean.TRUE;
096            } else if (currentToken == JsonToken.VALUE_FALSE) {
097                return Boolean.FALSE;
098            } else if (currentToken == JsonToken.VALUE_NULL) {
099                return null;
100            } else {
101                return text();
102            }
103        }
104    
105        @Override
106        public Object objectBytes() throws IOException {
107            JsonToken currentToken = parser.getCurrentToken();
108            if (currentToken == JsonToken.VALUE_STRING) {
109                return bytes();
110            } else if (currentToken == JsonToken.VALUE_NUMBER_INT || currentToken == JsonToken.VALUE_NUMBER_FLOAT) {
111                return parser.getNumberValue();
112            } else if (currentToken == JsonToken.VALUE_TRUE) {
113                return Boolean.TRUE;
114            } else if (currentToken == JsonToken.VALUE_FALSE) {
115                return Boolean.FALSE;
116            } else if (currentToken == JsonToken.VALUE_NULL) {
117                return null;
118            } else {
119                return bytes();
120            }
121        }
122    
123        @Override
124        public boolean hasTextCharacters() {
125            return parser.hasTextCharacters();
126        }
127    
128    
129        @Override
130        public char[] textCharacters() throws IOException {
131            return parser.getTextCharacters();
132        }
133    
134    
135        @Override
136        public int textLength() throws IOException {
137            return parser.getTextLength();
138        }
139    
140    
141        @Override
142        public int textOffset() throws IOException {
143            return parser.getTextOffset();
144        }
145    
146    
147        @Override
148        public Number numberValue() throws IOException {
149            return parser.getNumberValue();
150        }
151    
152    
153        @Override
154        public short doShortValue() throws IOException {
155            return parser.getShortValue();
156        }
157    
158    
159        @Override
160        public int doIntValue() throws IOException {
161            return parser.getIntValue();
162        }
163    
164    
165        @Override
166        public long doLongValue() throws IOException {
167            return parser.getLongValue();
168        }
169    
170    
171        @Override
172        public float doFloatValue() throws IOException {
173            return parser.getFloatValue();
174        }
175    
176    
177        @Override
178        public double doDoubleValue() throws IOException {
179            return parser.getDoubleValue();
180        }
181    
182    
183        @Override
184        public byte[] binaryValue() throws IOException {
185            return parser.getBinaryValue();
186        }
187    
188    
189        @Override
190        public void close() {
191            try {
192                parser.close();
193            } catch (IOException e) {
194                // ignore
195            }
196        }
197    
198        private NumberType convertNumberType(JsonParser.NumberType numberType) {
199            switch (numberType) {
200                case INT:
201                    return NumberType.INT;
202                case LONG:
203                    return NumberType.LONG;
204                case FLOAT:
205                    return NumberType.FLOAT;
206                case DOUBLE:
207                    return NumberType.DOUBLE;
208            }
209            throw new IllegalStateException("No matching token for number_type [" + numberType + "]");
210        }
211    
212        private Token convertToken(JsonToken token) {
213            if (token == null) {
214                return null;
215            }
216            switch (token) {
217                case FIELD_NAME:
218                    return Token.FIELD_NAME;
219                case VALUE_FALSE:
220                case VALUE_TRUE:
221                    return Token.VALUE_BOOLEAN;
222                case VALUE_STRING:
223                    return Token.VALUE_STRING;
224                case VALUE_NUMBER_INT:
225                case VALUE_NUMBER_FLOAT:
226                    return Token.VALUE_NUMBER;
227                case VALUE_NULL:
228                    return Token.VALUE_NULL;
229                case START_OBJECT:
230                    return Token.START_OBJECT;
231                case END_OBJECT:
232                    return Token.END_OBJECT;
233                case START_ARRAY:
234                    return Token.START_ARRAY;
235                case END_ARRAY:
236                    return Token.END_ARRAY;
237                case VALUE_EMBEDDED_OBJECT:
238                    return Token.VALUE_EMBEDDED_OBJECT;
239            }
240            throw new IllegalStateException("No matching token for json_token [" + token + "]");
241        }
242    }
