002package org.xbib.elasticsearch.common.xcontent.xml;
003
004import javax.xml.namespace.QName;
005
006/**
007 * XML parameters for XML XContent
008 */
009public class XmlXParams {
010
011    public final static QName DEFAULT_ROOT = new QName("http://elasticsearch.org/ns/1.0/", "root", "es");
012
013    private final QName root;
014
015    private XmlNamespaceContext namespaceContext;
016
017    public XmlXParams() {
018        this(null, null);
019    }
020
021    public XmlXParams(XmlNamespaceContext namespaceContext) {
022        this(null, namespaceContext);
023    }
024
025    public XmlXParams(QName root, XmlNamespaceContext namespaceContext) {
026        this.root = root != null ? root : DEFAULT_ROOT;
027        if (namespaceContext == null) {
028            namespaceContext = XmlNamespaceContext.getDefaultInstance();
029            namespaceContext.addNamespace(DEFAULT_ROOT.getPrefix(), DEFAULT_ROOT.getNamespaceURI());
030        } else {
031            namespaceContext.addNamespace(DEFAULT_ROOT.getPrefix(), DEFAULT_ROOT.getNamespaceURI());
032            this.namespaceContext = namespaceContext;
033        }
034    }
035
036    public QName getQName() {
037        return root;
038    }
039
040    public XmlNamespaceContext getNamespaceContext() {
041        return namespaceContext;
042    }
043
044}
