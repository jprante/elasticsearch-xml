020package org.xbib.elasticsearch.common.xcontent.xml;
021
022import javax.xml.namespace.QName;
023
024/**
025 * XML parameters for XML XContent
026 */
027public class XmlXParams {
028
029    private final static QName defaultRoot = new QName("http://elasticsearch.org/ns/1.0/", "root", "es");
030
031    private final QName root;
032
033    private final XmlNamespaceContext namespaceContext;
034
035    public XmlXParams() {
036        this(null, null);
037    }
038
039    public XmlXParams(XmlNamespaceContext namespaceContext) {
040        this(null, namespaceContext);
041    }
042
043    public XmlXParams(QName root, XmlNamespaceContext namespaceContext) {
044        this.root = root != null ? root : defaultRoot;
045        if (namespaceContext == null) {
046            this.namespaceContext = getDefaultNamespaceContext();
047        } else {
048            namespaceContext.addNamespace(defaultRoot.getPrefix(), defaultRoot.getNamespaceURI());
049            this.namespaceContext = namespaceContext;
050        }
051    }
052
053    public QName getQName() {
054        return root;
055    }
056
057    public XmlNamespaceContext getNamespaceContext() {
058        return namespaceContext;
059    }
060
061    public static XmlNamespaceContext getDefaultNamespaceContext() {
062        XmlNamespaceContext namespaceContext = XmlNamespaceContext.newInstance();
063        namespaceContext.addNamespace(defaultRoot.getPrefix(), defaultRoot.getNamespaceURI());
064        return namespaceContext;
065    }
066
067}
