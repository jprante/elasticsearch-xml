003    package org.xbib.elasticsearch.common.xcontent;
004    
005    import org.elasticsearch.common.xcontent.XContent;
006    import org.elasticsearch.common.xcontent.json.JsonXContent;
007    import org.elasticsearch.common.xcontent.smile.SmileXContent;
008    import org.elasticsearch.common.xcontent.yaml.YamlXContent;
009    
010    import org.xbib.elasticsearch.common.xcontent.xml.XmlXContent;
011    
012    /**
013     * The content type of {@link org.elasticsearch.common.xcontent.XContent}.
014     */
015    public enum XmlXContentType {
016    
017        /**
018         * A JSON based content type.
019         */
020        JSON(0) {
021            @Override
022            public String restContentType() {
023                return "application/json; charset=UTF-8";
024            }
025    
026            @Override
027            public String shortName() {
028                return "json";
029            }
030    
031            @Override
032            public XContent xContent() {
033                return JsonXContent.jsonXContent;
034            }
035        },
036        /**
037         * The jackson based smile binary format. Fast and compact binary format.
038         */
039        SMILE(1) {
040            @Override
041            public String restContentType() {
042                return "application/smile";
043            }
044    
045            @Override
046            public String shortName() {
047                return "smile";
048            }
049    
050            @Override
051            public XContent xContent() {
052                return SmileXContent.smileXContent;
053            }
054        },
055        /**
056         * The YAML format.
057         */
058        YAML(2) {
059            @Override
060            public String restContentType() {
061                return "application/yaml";
062            }
063    
064            @Override
065            public String shortName() {
066                return "yaml";
067            }
068    
069            @Override
070            public XContent xContent() {
071                return YamlXContent.yamlXContent;
072            }
073        },
074        /**
075         * The XML format.
076         */
077        XML(3) {
078            @Override
079            public String restContentType() {
080                return "application/xml";
081            }
082    
083            @Override
084            public String shortName() {
085                return "xml";
086            }
087    
088            @Override
089            public XContent xContent() {
090                return XmlXContent.xmlXContent();
091            }
092        };
093    
094        public static XmlXContentType fromRestContentType(String contentType) {
095            if (contentType == null) {
096                return null;
097            }
098            if ("application/json".equals(contentType) || "json".equalsIgnoreCase(contentType)) {
099                return JSON;
100            }
101            if ("application/smile".equals(contentType) || "smile".equalsIgnoreCase(contentType)) {
102                return SMILE;
103            }
104            if ("application/yaml".equals(contentType) || "yaml".equalsIgnoreCase(contentType)) {
105                return YAML;
106            }
107            if ("application/xml".equals(contentType) || "xml".equalsIgnoreCase(contentType)) {
108                return XML;
109            }
110            return null;
111        }
112    
113        private int index;
114    
115        XmlXContentType(int index) {
116            this.index = index;
117        }
118    
119        public int index() {
120            return index;
121        }
122    
123        public abstract String restContentType();
124    
125        public abstract String shortName();
126    
127        public abstract XContent xContent();
128    }
