020package org.xbib.elasticsearch.common.xcontent;
021
022import org.elasticsearch.common.xcontent.XContent;
023import org.elasticsearch.common.xcontent.json.JsonXContent;
024import org.elasticsearch.common.xcontent.smile.SmileXContent;
025import org.elasticsearch.common.xcontent.yaml.YamlXContent;
026
027import org.xbib.elasticsearch.common.xcontent.xml.XmlXContent;
028
029/**
030 * The content type of {@link org.elasticsearch.common.xcontent.XContent}.
031 */
032public enum XmlXContentType {
033
034    /**
035     * A JSON based content type.
036     */
037    JSON(0) {
038        @Override
039        public String restContentType() {
040            return "application/json; charset=UTF-8";
041        }
042
043        @Override
044        public String shortName() {
045            return "json";
046        }
047
048        @Override
049        public XContent xContent() {
050            return JsonXContent.jsonXContent;
051        }
052    },
053    /**
054     * The jackson based smile binary format. Fast and compact binary format.
055     */
056    SMILE(1) {
057        @Override
058        public String restContentType() {
059            return "application/smile";
060        }
061
062        @Override
063        public String shortName() {
064            return "smile";
065        }
066
067        @Override
068        public XContent xContent() {
069            return SmileXContent.smileXContent;
070        }
071    },
072    /**
073     * The YAML format.
074     */
075    YAML(2) {
076        @Override
077        public String restContentType() {
078            return "application/yaml";
079        }
080
081        @Override
082        public String shortName() {
083            return "yaml";
084        }
085
086        @Override
087        public XContent xContent() {
088            return YamlXContent.yamlXContent;
089        }
090    },
091    /**
092     * The XML format.
093     */
094    XML(3) {
095        @Override
096        public String restContentType() {
097            return "application/xml";
098        }
099
100        @Override
101        public String shortName() {
102            return "xml";
103        }
104
105        @Override
106        public XContent xContent() {
107            return XmlXContent.xmlXContent();
108        }
109    };
110
111    public static XmlXContentType fromRestContentType(String contentType) {
112        if (contentType == null) {
113            return null;
114        }
115        if ("application/json".equals(contentType) || "json".equalsIgnoreCase(contentType)) {
116            return JSON;
117        }
118        if ("application/smile".equals(contentType) || "smile".equalsIgnoreCase(contentType)) {
119            return SMILE;
120        }
121        if ("application/yaml".equals(contentType) || "yaml".equalsIgnoreCase(contentType)) {
122            return YAML;
123        }
124        if ("application/xml".equals(contentType) || "xml".equalsIgnoreCase(contentType)) {
125            return XML;
126        }
127        return null;
128    }
129
130    private int index;
131
132    XmlXContentType(int index) {
133        this.index = index;
134    }
135
136    public int index() {
137        return index;
138    }
139
140    public abstract String restContentType();
141
142    public abstract String shortName();
143
144    public abstract XContent xContent();
145}
