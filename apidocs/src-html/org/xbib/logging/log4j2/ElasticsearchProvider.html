016package org.xbib.logging.log4j2;
017
018import org.apache.logging.log4j.Logger;
019import org.apache.logging.log4j.core.config.plugins.Plugin;
020import org.apache.logging.log4j.core.config.plugins.PluginAttribute;
021import org.apache.logging.log4j.core.config.plugins.PluginFactory;
022import org.apache.logging.log4j.nosql.appender.NoSqlProvider;
023import org.apache.logging.log4j.status.StatusLogger;
024import org.elasticsearch.client.transport.TransportClient;
025import org.elasticsearch.common.settings.Settings;
026import org.elasticsearch.common.transport.InetSocketTransportAddress;
027import org.elasticsearch.common.unit.ByteSizeValue;
028import org.elasticsearch.common.unit.TimeValue;
029
030import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
031
032@Plugin(name = "Elasticsearch", category = "Core", printObject = true)
033public class ElasticsearchProvider implements NoSqlProvider<ElasticsearchConnection> {
034
035    private static final Logger logger = StatusLogger.getLogger();
036
037    private final BulkTransportClient client;
038
039    private final String description;
040
041    private ElasticsearchProvider(final BulkTransportClient client, final String description) {
042        this.client = client;
043        this.description = "elasticsearch{ " + description + " }";
044    }
045
046    @Override
047    public ElasticsearchConnection getConnection() {
048        return new ElasticsearchConnection(client);
049    }
050
051    @Override
052    public String toString() {
053        return description;
054    }
055
056    /**
057     * Factory method for creating an Elasticsearch provider within the plugin manager.
058     *
059     * @param cluster The name of the Elasticsearch cluster to which log event documents will be written.
060     * @param host    The host name an Elasticsearch server node of the cluster, defaults to localhost.
061     * @param port    The port that Elasticsearch is listening on, defaults to 9300
062     * @param index   The index that Elasticsearch shall use for indexing
063     * @param type    The type of the index Elasticsearch shall use for indexing
064     * @return a new Elasticsearch provider
065     */
066    @PluginFactory
067    public static ElasticsearchProvider createNoSqlProvider(
068            @PluginAttribute("cluster") String cluster,
069            @PluginAttribute("host") String host,
070            @PluginAttribute("port") Integer port,
071            @PluginAttribute("index") String index,
072            @PluginAttribute("type") String type,
073            @PluginAttribute("sniff") Boolean sniff,
074            @PluginAttribute("timeout") String timeout,
075            @PluginAttribute("maxActionsPerBulkRequest") Integer maxActionsPerBulkRequest,
076            @PluginAttribute("maxConcurrentBulkRequests") Integer maxConcurrentBulkRequests,
077            @PluginAttribute("maxVolumePerBulkRequest") String maxVolumePerBulkRequest,
078            @PluginAttribute("flushInterval") String flushInterval) {
079
080        if (cluster == null || cluster.isEmpty()) {
081            cluster = "elasticsearch";
082        }
083        if (host == null || host.isEmpty()) {
084            host = "localhost";
085        }
086        if (port == null || port == 0) {
087            port = 9300;
088        }
089        if (index == null || index.isEmpty()) {
090            index = "log4j2";
091        }
092        if (type == null || type.isEmpty()) {
093            type = "log4j2";
094        }
095        if (sniff == null) {
096            sniff = false;
097        }
098        if (timeout == null || timeout.isEmpty()) {
099            timeout = "30s";
100        }
101        if (maxActionsPerBulkRequest == null) {
102            maxActionsPerBulkRequest = 1000;
103        }
104        if (maxConcurrentBulkRequests == null) {
105            maxConcurrentBulkRequests = 2 * Runtime.getRuntime().availableProcessors();
106        }
107        if (maxVolumePerBulkRequest == null || maxVolumePerBulkRequest.isEmpty()) {
108            maxVolumePerBulkRequest = "10m";
109        }
110
111        Settings settings = settingsBuilder()
112                .put("cluster.name", cluster)
113                .put("network.server", false)
114                .put("node.client", true)
115                .put("client.transport.sniff", sniff)
116                .put("client.transport.ping_timeout", timeout)
117                .put("client.transport.ignore_cluster_name", false)
118                .put("client.transport.nodes_sampler_interval", "30s")
119                .build();
120
121        TransportClient client = new TransportClient(settings, false)
122                .addTransportAddress(new InetSocketTransportAddress(host, port));
123        if (client.connectedNodes().isEmpty()) {
124            logger.error("unable to connect to Elasticsearch cluster");
125            return null;
126        }
127        String description = "cluster=" + cluster + ",host=" + host + ",port=" + port + ",index=" + index + ",type=" + type;
128        BulkTransportClient bulkTransportClient = new BulkTransportClient(client, index, type,
129                maxActionsPerBulkRequest, maxConcurrentBulkRequests,
130                ByteSizeValue.parseBytesSizeValue(maxVolumePerBulkRequest),
131                TimeValue.parseTimeValue(flushInterval, TimeValue.timeValueSeconds(30)));
132
133        return new ElasticsearchProvider(bulkTransportClient, description);
134    }
135
