016package org.xbib.logging.log4j2;
017
018import org.elasticsearch.ElasticsearchIllegalStateException;
019import org.elasticsearch.action.bulk.BulkProcessor;
020import org.elasticsearch.action.bulk.BulkRequest;
021import org.elasticsearch.action.bulk.BulkResponse;
022import org.elasticsearch.action.index.IndexRequest;
023import org.elasticsearch.client.Client;
024import org.elasticsearch.common.unit.ByteSizeValue;
025import org.elasticsearch.common.unit.TimeValue;
026
027import java.util.Map;
028
029public class BulkTransportClient {
030
031    private final Client client;
032
033    private final BulkProcessor bulkProcessor;
034
035    private final String index;
036
037    private final String type;
038
039    private volatile boolean closed = false;
040
041    public BulkTransportClient(Client client, String index, String type,
042                               int maxActionsPerBulkRequest,
043                               int maxConcurrentBulkRequests,
044                               ByteSizeValue maxVolumePerBulkRequest,
045                               TimeValue flushInterval) {
046        this.client = client;
047        this.index = index;
048        this.type = type;
049        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
050            @Override
051            public void beforeBulk(long executionId, BulkRequest request) {
052            }
053
054            @Override
055            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
056            }
057
058            @Override
059            public void afterBulk(long executionId, BulkRequest requst, Throwable failure) {
060                closed = true;
061            }
062        };
063        BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
064                .setBulkActions(maxActionsPerBulkRequest)
065                .setConcurrentRequests(maxConcurrentBulkRequests)
066                .setFlushInterval(flushInterval);
067        if (maxVolumePerBulkRequest != null) {
068            builder.setBulkSize(maxVolumePerBulkRequest);
069        }
070        this.bulkProcessor = builder.build();
071        this.closed = false;
072    }
073
074    public BulkTransportClient index(Map<String, Object> source) {
075        if (closed) {
076            throw new ElasticsearchIllegalStateException("client is closed");
077        }
078        try {
079            bulkProcessor.add(new IndexRequest(index).type(type).create(false).source(source));
080        } catch (Exception e) {
081            closed = true;
082        }
083        return this;
084    }
085
086    public void close() {
087        bulkProcessor.close();
088        client.close();
089    }
090}
