016package org.xbib.logging.log4j2;
017
018import org.apache.logging.log4j.core.appender.AppenderLoggingException;
019import org.apache.logging.log4j.nosql.appender.NoSqlConnection;
020import org.apache.logging.log4j.nosql.appender.NoSqlObject;
021
022import java.util.Map;
023import java.util.concurrent.atomic.AtomicBoolean;
024
025public class ElasticsearchConnection implements NoSqlConnection<Map<String, Object>, ElasticsearchObject> {
026
027    private final BulkTransportClient client;
028
029    private final AtomicBoolean closed = new AtomicBoolean(false);
030
031    public ElasticsearchConnection(final BulkTransportClient client) {
032        this.client = client;
033    }
034
035    @Override
036    public ElasticsearchObject createObject() {
037        return new ElasticsearchObject();
038    }
039
040    @Override
041    public ElasticsearchObject[] createList(final int length) {
042        return new ElasticsearchObject[length];
043    }
044
045    @Override
046    public void insertObject(final NoSqlObject<Map<String, Object>> object) {
047        try {
048            client.index(object.unwrap());
049        } catch (Exception e) {
050            throw new AppenderLoggingException("failed to write log event to Elasticsearch: " + e.getMessage(), e);
051        }
052    }
053
054    @Override
055    public void close() {
056        if (closed.compareAndSet(false, true)) {
057            client.close();
058        }
059    }
060
061    @Override
062    public boolean isClosed() {
063        return closed.get();
064    }
065}
