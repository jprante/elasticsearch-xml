001package org.apache.logging.log4j.nosql.appender;
002
003import org.junit.Test;
004
005import static org.easymock.EasyMock.createStrictMock;
006import static org.easymock.EasyMock.replay;
007import static org.easymock.EasyMock.verify;
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertNotNull;
010import static org.junit.Assert.assertNull;
011
012public class NoSqlAppenderTest {
013    @Test
014    public void testNoProvider() {
015        final NoSqlAppender appender = NoSqlAppender.createAppender("myName01", null, null, null, null);
016        assertNull("The appender should be null.", appender);
017    }
018
019    @Test
020    public void testProvider() {
021        @SuppressWarnings("unchecked")
022        final NoSqlProvider<?> provider = createStrictMock(NoSqlProvider.class);
023        replay(provider);
024        final NoSqlAppender appender = NoSqlAppender.createAppender("myName01", null, null, null, provider);
025        assertNotNull("The appender should not be null.", appender);
026        assertEquals("The toString value is not correct.",
027                "myName01{ manager=noSqlManager{ description=myName01, bufferSize=0, provider=" + provider + " } }",
028                appender.toString());
029        appender.stop();
030        verify(provider);
031    }
032
033    @Test
034    public void testProviderBuffer() {
035        @SuppressWarnings("unchecked")
036        final NoSqlProvider<?> provider = createStrictMock(NoSqlProvider.class);
037        replay(provider);
038        final NoSqlAppender appender = NoSqlAppender.createAppender("anotherName02", null, null, "25", provider);
039        assertNotNull("The appender should not be null.", appender);
040        assertEquals("The toString value is not correct.",
041                "anotherName02{ manager=noSqlManager{ description=anotherName02, bufferSize=25, provider=" + provider
042                        + " } }", appender.toString());
043        appender.stop();
044        verify(provider);
045    }
046}
