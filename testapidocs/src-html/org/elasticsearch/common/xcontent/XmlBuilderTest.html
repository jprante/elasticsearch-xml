059        @Test
060        public void testXmlCustomNamespaces() throws Exception {
061            QName root = new QName("http://elasticsearch.org/ns/1.0/", "result", "");
062            XmlNamespaceContext context = XmlNamespaceContext.getDefaultInstance();
063            context.addNamespace("abc", "http://localhost");
064            XmlXParams params = new XmlXParams(root, context);
065            XmlXContentBuilder builder = xmlBuilder(params);
066            builder.startObject()
067                    .field("abc:creator", "John Doe")
068                    .endObject();
069            assertEquals(
070                    builder.string(),
071                    "<result xmlns=\"http://elasticsearch.org/ns/1.0/\" xmlns:abc=\"http://localhost\" " +
072                            "xmlns:atom=\"http://www.w3.org/2005/Atom\" " +
073                            "xmlns:dc=\"http://purl.org/dc/elements/1.1/\" " +
074                            "xmlns:dcterms=\"http://purl.org/dc/terms/\" " +
075                            "xmlns:es=\"http://elasticsearch.org/ns/1.0/\" " +
076                            "xmlns:foaf=\"http://xmlns.com/foaf/0.1/\" xmlns:owl=\"http://www.w3.org/2002/07/owl#\" xmlns:rdf=\"http://www.w3.org/1999/02/22-rdf-syntax-ns#\" xmlns:rdfs=\"http://www.w3.org/2000/01/rdf-schema#\" xmlns:xalan=\"http://xml.apache.org/xslt\" xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\"><abc:creator>John Doe</abc:creator></result>"

080        @Test
081        public void testXmlObject() throws Exception {
082            XmlXParams params = new XmlXParams();
083            XmlXContentBuilder builder = xmlBuilder(params);
084            builder.startObject()
085                    .startObject("author")
086                    .field("creator", "John Doe")
087                    .field("role", "writer")
088                    .endObject()
089                    .startObject("author")
090                    .field("creator", "Joe Smith")
091                    .field("role", "illustrator")
092                    .endObject()
093                    .endObject();
094            assertEquals(
095                    builder.string(),
096                    "<root xmlns=\"http://elasticsearch.org/ns/1.0/\"><author><creator>John Doe</creator><role>writer</role></author><author><creator>Joe Smith</creator><role>illustrator</role></author></root>"
097            );

101        @Test
102        public void testXmlAttributes() throws Exception {
103            XmlNamespaceContext namespaceContext = XmlNamespaceContext.newInstance();
104            namespaceContext.addNamespace("es", "http://elasticsearch.org/ns/1.0/");
105            XmlXParams params = new XmlXParams(namespaceContext);
106            XmlXContentBuilder builder = xmlBuilder(params);
107            builder.startObject()
108                    .startObject("author")
109                    .field("@name", "John Doe")
110                    .field("@id", 1)
111                    .endObject()
112                    .endObject();
113            assertEquals(
114                    builder.string(),
115                    "<root xmlns=\"http://elasticsearch.org/ns/1.0/\" xmlns:es=\"http://elasticsearch.org/ns/1.0/\"><author es:name=\"John Doe\" es:id=\"1\"/></root>"
116            );
117        }

154        @Test
155        public void testParseJson() throws Exception {
156            XmlNamespaceContext context = XmlNamespaceContext.getDefaultInstance();
157            context.addNamespace("bib","info:srw/cql-context-set/1/bib-v1/");
158            context.addNamespace("abc", "http://localhost/");
159            context.addNamespace("xbib", "http://xbib.org/");
160            context.addNamespace("lia", "http://xbib.org/namespaces/lia/");
161            XmlXParams params = new XmlXParams(context);
162            InputStream in = getClass().getResourceAsStream("/test.json");
163            ByteArrayOutputStream out = new ByteArrayOutputStream();
164            Streams.copy(in, out);
165            byte[] buf = out.toByteArray();
166            String s = XmlXContentHelper.convertToXml(params, buf, 0, buf.length, false);
167            assertEquals(53194, s.length());
168        }
