002package org.xbib.elasticsearch.action.simple;
003
004import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
005import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.common.logging.ESLogger;
008import org.elasticsearch.common.logging.ESLoggerFactory;
009import org.elasticsearch.common.network.NetworkUtils;
010import org.elasticsearch.common.settings.ImmutableSettings;
011import org.elasticsearch.common.settings.Settings;
012import org.elasticsearch.common.transport.InetSocketTransportAddress;
013import org.elasticsearch.node.Node;
014import org.junit.After;
015import org.junit.Before;
016
017import java.util.Map;
018import java.util.concurrent.atomic.AtomicInteger;
019
020import static org.elasticsearch.common.collect.Maps.newHashMap;
021import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
022import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
023
024public abstract class AbstractNodeTestHelper {
025
026    protected final static ESLogger logger = ESLoggerFactory.getLogger("test");
027
028    private Map<String, Node> nodes = newHashMap();
029
030    private Map<String, Client> clients = newHashMap();
031
032    private AtomicInteger counter = new AtomicInteger();
033
034    private String cluster;
035
036    private String host;
037
038    private int port;
039
040    protected void setClusterName() {
041        this.cluster = "test-cluster-" + NetworkUtils.getLocalAddress().getHostName() + "-" + counter.incrementAndGet();
042    }
043
044    protected String getClusterName() {
045        return cluster;
046    }
047
048    protected String getHost() {
049        return host;
050    }
051
052    protected int getPort() {
053        return port;
054    }
055
056    protected Settings getNodeSettings() {
057        return ImmutableSettings
058                .settingsBuilder()
059                .put("cluster.name", getClusterName())
060                .put("index.number_of_shards", 2)
061                .put("index.number_of_replicas", 1)
062                .put("cluster.routing.schedule", "50ms")
063                .put("gateway.type", "none")
064                .put("index.store.type", "ram")
065                .put("http.enabled", false)
066                .put("discovery.zen.multicast.enabled", false)
067                .put("threadpool.bulk.queue_size", 200)
068                .build();
069    }
070
071    @Before
072    public void startNodes() throws Exception {
073        setClusterName();
074        startNode("1");
075        // find node address
076        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
077        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
078        Object obj = response.iterator().next().getTransport().getAddress().publishAddress();
079        if (obj instanceof InetSocketTransportAddress) {
080            InetSocketTransportAddress address = (InetSocketTransportAddress) obj;
081            host = address.address().getHostName();
082            port = address.address().getPort();
083        }
084    }
085
086    @After
087    public void stopNodes() throws Exception {
088        closeAllNodes();
089    }
090
091    protected Node startNode(String id) {
092        return buildNode(id).start();
093    }
094
095    private Node buildNode(String id) {
096        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
097        Settings finalSettings = settingsBuilder()
098                .loadFromClasspath(settingsSource)
099                .put(getNodeSettings())
100                .put("name", id)
101                .build();
102        Node node = nodeBuilder().settings(finalSettings).build();
103        Client client = node.client();
104        nodes.put(id, node);
105        clients.put(id, client);
106        return node;
107    }
108
109    protected void stopNode(String id) {
110        Client client = clients.remove(id);
111        if (client != null) {
112            client.close();
113        }
114        Node node = nodes.remove(id);
115        if (node != null) {
116            node.close();
117        }
118    }
119
120    public Client client(String id) {
121        return clients.get(id);
122    }
123
124    public void closeAllNodes() {
125        for (Client client : clients.values()) {
126            client.close();
127        }
128        clients.clear();
129        for (Node node : nodes.values()) {
130            if (node != null) {
131                node.close();
132            }
133        }
134        nodes.clear();
135    }
136
137}
