001package org.xbib.elasticsearch.action.simple;
002
003import org.elasticsearch.action.index.IndexRequest;
004import org.elasticsearch.action.search.SearchResponse;
005import org.elasticsearch.client.Client;
006import org.junit.Test;
007
008public class SimpleActionTest extends AbstractNodeTestHelper {
009
010    @Test
011    public void testSimpleAction() throws NullPointerException {
012        Client client = client("1");
013
014        // index a test document
015        IndexRequest indexRequest = new IndexRequest().index("test").type("test").id("test")
016                .source("Hello", "World")
017                .refresh(true);
018        client.index(indexRequest).actionGet();
019
020        // execute simple action (match all query)
021        SimpleRequestBuilder builder = new SimpleRequestBuilder(client);
022        SearchResponse response = client.execute(SimpleAction.INSTANCE, builder.request()).actionGet();
023        logger.info("result of simple action = {}", response.toString());
024    }
025
026}
