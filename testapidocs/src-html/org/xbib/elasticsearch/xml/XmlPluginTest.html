001package org.xbib.elasticsearch.xml;
002
003import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
004import org.elasticsearch.action.admin.indices.refresh.RefreshRequest;
005import org.elasticsearch.action.delete.DeleteRequest;
006import org.elasticsearch.action.index.IndexRequest;
007import org.elasticsearch.client.Client;
008import org.elasticsearch.common.settings.ImmutableSettings;
009import org.elasticsearch.common.settings.Settings;
010import org.testng.annotations.AfterClass;
011import org.testng.annotations.BeforeClass;
012import org.testng.annotations.Test;
013import org.xbib.elasticsearch.integration.AbstractNodesTests;
014
015import java.io.BufferedReader;
016import java.io.InputStreamReader;
017import java.net.URL;
018import java.util.Random;
019
020import static org.testng.AssertJUnit.assertTrue;
021
022public class XmlPluginTest extends AbstractNodesTests {
023
024    private Client client;
025
026    @BeforeClass
027    public void createNodes() throws Exception {
028        Settings settings = ImmutableSettings.settingsBuilder()
029                .put("index.number_of_shards", numberOfShards())
030                .put("index.number_of_replicas", 0)
031                .build();
032        for (int i = 0; i < numberOfNodes(); i++) {
033            startNode("node" + i, settings);
034        }
035        client = getClient();
036    }
037
038    protected int numberOfShards() {
039        return 1;
040    }
041
042    protected int numberOfNodes() {
043        return 1;
044    }
045
046    @AfterClass
047    public void closeNodes() {
048        client.close();
049        closeAllNodes();
050    }
051
052    protected Client getClient() {
053        return client("node0");
054    }
055
056    @Test
057    public void testHealthResponse() throws Exception {
058        URL url = new URL(getHttpAddressOfNode("node0").toURL(), "/_cluster/health?xml");
059        BufferedReader reader = new BufferedReader(new InputStreamReader(url.openStream()));
060        String line;
061        if ((line = reader.readLine()) != null) {
062            assertTrue(line.startsWith("<root xmlns=\"http://elasticsearch.org/ns/1.0/\">"));
063            assertTrue(line.endsWith("</root>"));
064        }
065        reader.close();
066    }
067
068    @Test
069    public void testBigAndFatResponse() throws Exception {
070        Client client = getClient();
071        for (int i = 0; i < 10000; i++) {
072            client.index(new IndexRequest("test", "test", Integer.toString(i))
073                    .source("{\"random\":\""+randomString(32)+ " " + randomString(32) + "\"}")).actionGet();
074        }
075        client.admin().indices().refresh(new RefreshRequest("test")).actionGet();
076        URL url = new URL(getHttpAddressOfNode("node0").toURL(), "/test/test/_search?xml&pretty&size=10000");
077        BufferedReader reader = new BufferedReader(new InputStreamReader(url.openStream()));
078        int count = 0;
079        String line;
080        while ((line = reader.readLine()) != null) {
081            count += line.length();
082        }
083        assertTrue(count >= 2309156);
084        reader.close();
085        client.admin().indices().delete(new DeleteIndexRequest("test"));
086    }
087
088    private static Random random = new Random();
089
090    private static char[] numbersAndLetters = ("0123456789abcdefghijklmnopqrstuvwxyz").toCharArray();
091
092    protected String randomString(int len) {
093        final char[] buf = new char[len];
094        final int n = numbersAndLetters.length - 1;
095        for (int i = 0; i < buf.length; i++) {
096            buf[i] = numbersAndLetters[random.nextInt(n)];
097        }
098        return new String(buf);
099    }
100}
